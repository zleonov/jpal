044/**
045 * A {@code List} optimized for efficient <a target="_blank" href="http://en.wikipedia.org/wiki/Random_access">random
046 * access</a> insertion and removal operations. Implements all optional list operations, and permits all elements,
047 * including {@code null}.
048 * <p>
049 * The iterators obtained from the {@link #iterator()} and {@link #listIterator()} methods are <i>fail-fast</i>.
050 * Attempts to modify the elements in this list at any time after an iterator is created, in any way except through the
051 * iterator's own remove method, will result in a {@code ConcurrentModificationException}.
052 * <p>
053 * This list is not <i>thread-safe</i>. If multiple threads modify this list concurrently it must be
054 * {@link Collections#synchronizedList(List) synchronized} externally.
055 * <p>
056 * This class implements an array-based <a target="_blank" href="http://en.wikipedia.org/wiki/Skip_list">skip list</a>
057 * modified to provide logarithmic running time for linear list operations (e.g. insert an element at the i<i>th</i>
058 * index). Linear list operations are sometimes referred to as rank operations.
059 * <p>
060 * Invented by <a target="_blank" href="http://www.cs.umd.edu/~pugh/">Bill Pugh</a> in 1990, A skip list is a
061 * probabilistic data structure for maintaining items in sorted order. Strictly speaking it is impossible to make any
062 * hard guarantees regarding the worst-case performance of this class. Practical performance is <i>expected</i> to be
063 * logarithmic with an extremely high degree of probability as the list grows.
064 * <p>
065 * The following table summarizes the <i>big-O</i> performance of this class compared to an {@link ArrayList} and a
066 * {@link LinkedList} (where n is the size of this list and <i>m</i> is the size of the specified collection which is
067 * iterable in linear time):
068 * 
069 * <pre>
070 * <table border cellpadding="3" cellspacing="1" style="width:450px;">
071 *   <tr>
072 *     <th rowspan="2">Method</th>
073 *     <th colspan="3">Running Time</th>
074 *   </tr>
075 *   <tr>
076 *     <td style="text-align:center;"><b>RankList</b><br>(<i>expected</i>)</td>
077 *     <td style="text-align:center;"><b>ArrayList</b><br>(<i>amortized</i>)</td>
078 *     <td style="text-align:center;"><b>LinkedList</b><br>(<i>worst-case</i>)</td>
079 *   </tr>
080 *   <tr>
081 *     <td>{@link RankList#addAll(Collection) addAll(Collection)}</td>
082 *     <td style="text-align:center;" bgcolor="FFCC99"><i>O(m log n)</i></td>
083 *     <td style="text-align:center;" bgcolor="FFCCCC"><i>O(m)</i></td>
084 *     <td style="text-align:center;" bgcolor="9999CC"><i>O(m)</i></td>
085 *   </tr>
086 *   <tr>
087 *     <td>{@link #containsAll(Collection) containsAll(Collection)}<br>{@link #removeAll(Collection) removeAll(Collection)}<br>{@link #retainAll(Collection) retainAll(Collection)}</td>
088 *     <td style="text-align:center;" colspan="3"><i>O(m * n)</i></td>
089 *   </tr>
090 *     <td>{@link #add(Object) add(E)}</td>
091 *     <td style="text-align:center;" rowspan="9" bgcolor="FFCC99"><i>O(log n)</i></td>
092 *     <td style="text-align:center;" rowspan="4" bgcolor="FFCCCC"><i>O(1)</i></td>
093 *     <td style="text-align:center;" bgcolor="9999CC"><i>O(1)</i></td>
094 *   </tr>
095 *   <tr>
096 *     <td>{@link #get(int)}</td>
097 *     <td style="text-align:center;" rowspan="5" bgcolor="9999CC"><i>O(n)</i></td>
098 *   </tr>
099 *   <tr>
100 *     <td>{@link #set(int, Object) set(int, E)}</td>
101 *   </tr>
102 *   <tr>
103 *     <td>{@link #listIterator(int)}</td>
104 *   </tr>
105 *   <tr>
106 *     <td>{@link #add(int, Object) add(int, E)}</td>
107 *     <td style="text-align:center;" rowspan="5" bgcolor="FFCCCC"><i>O(n)</i></td>
108 *   </tr>
109 *   <tr>
110 *     <td>{@link #remove(int)}</td>
111 *   </tr>
112 *   <tr>
113 *     <td>{@link ListIterator#add(Object) ListIterator.add(E)}</td>
114 *     <td style="text-align:center;" rowspan="3" bgcolor="9999CC"><i>O(1)</i></td>
115 *   </tr>
116 *   <tr>
117 *     <td>{@link Iterator#remove()}</td>
118 *   </tr>
119 *   <tr>
120 *     <td>{@link ListIterator#remove()}</td>
121 *   </tr>
122 *   <tr>
123 *     <td>{@link ListIterator#set(Object) ListIterator.set(E)}<br>{@link #clear()}</td>
124 *     <td style="text-align:center;" colspan="3"><i>O(1)</i></td>
125 *   </tr>
126 *   <tr>
127 *     <td>{@link #contains(Object) contains(Object)}<br>{@link #indexOf(Object) indexOf(Object)}<br>{@link #lastIndexOf(Object)}<br>{@link #remove(Object) remove(Object)}</td>
128 *     <td style="text-align:center;" rowspan="4" colspan="3"><i>O(n)</i></td>
129 *   </tr>
130 * </table>
131 * </pre>
132 * 
133 * The {@code subList} views exhibit identical time complexity, with the exception of the {@code clear()} operation
134 * which runs in linear time proportional to the size of the view.
135 * 
136 * @author Zhenya Leonov
137 * @param <E> the type of elements maintained by this list
138 */
139public final class RankList<E> extends AbstractList<E> implements List<E>, RandomAccess, Serializable, Cloneable {
140
141    private static final long   serialVersionUID = 1L;
142    private static final double P                = .5;
143    private static final int    MAX_LEVEL        = 32;
144    private transient int       size             = 0;
145    private transient int       level            = 1;
146    private transient Random    random           = ThreadLocalRandom.current();
147    private transient Node<E>   head             = new Node<E>(null, MAX_LEVEL);
148
149    private RankList() {
150        for (int i = 0; i < MAX_LEVEL; i++) {
151            head.next[i] = head;
152            head.dist[i] = 1;
153        }
154        head.prev = head;
155    }
156
157    /**
158     * Creates a new {@code RankList}.
159     *
160     * @param <E> the element type
161     * @return a new {@code RankList}
162     */
163    public static <E> RankList<E> create() {
164        return new RankList<E>();
165    }
166
167    /**
168     * Creates a new {@code RankList} containing the elements of the specified {@code Iterable}.
169     * 
170     * @param <E>      the element type
171     * @param elements the iterable whose elements are to be placed into the list
172     * @return a new {@code RankList} containing the elements of the specified iterable
173     */
174    public static <E> RankList<E> create(final Iterable<? extends E> elements) {
175        checkNotNull(elements, "elements == null");
176        RankList<E> list = new RankList<E>();
177        Iterables.addAll(list, elements);
178        return list;
179    }
180
181    @Override
182    public Iterator<E> iterator() {
183        return listIterator();
184    }
185
186    @Override
187    public int size() {
188        return size;
189    }
190
191    @Override
192    public void add(int index, E element) {
193        checkPositionIndex(index, size);
194        Node<E>       curr     = head;
195        int           pos      = 0;
196        final int     newLevel = randomLevel();
197        final Node<E> newNode  = new Node<E>(element, newLevel);
198        if (newLevel > level) {
199            for (int i = level; i < newLevel; i++)
200                head.dist[i] = size + 1;
201            level = newLevel;
202        }
203        for (int i = level - 1; i >= 0; i--) {
204            while (pos + curr.dist[i] <= index) {
205                pos  += curr.dist[i];
206                curr  = curr.next[i];
207            }
208            if (i > newLevel - 1)
209                curr.dist[i]++;
210            else {
211                newNode.next[i] = curr.next[i];
212                curr.next[i]    = newNode;
213                newNode.dist[i] = pos + curr.dist[i] - index;
214                curr.dist[i]    = index + 1 - pos;
215            }
216        }
217        newNode.prev         = curr;
218        newNode.next[0].prev = newNode;
219        modCount++;
220        size++;
221    }
222
223    @Override
224    public E get(int index) {
225        checkElementIndex(index, size);
226        return search(index).element;
227
228    }
229
230    @Override
231    public int lastIndexOf(Object o) {
232        Node<E> node = head;
233        for (int i = size - 1; i >= 0; i--) {
234            node = node.prev;
235            if (Objects.equal(node.element, o))
236                return i;
237        }
238        return -1;
239    }
240
241    @Override
242    public ListIterator<E> listIterator() {
243        return listIterator(0);
244    }
245
246    @Override
247    public ListIterator<E> listIterator(final int index) {
248        checkPositionIndex(index, size);
249        return new ListIterator<E>() {
250            private Node<E> next             = (index == 0) ? head.next[0] : search(index);
251            private Node<E> last             = null;
252            private int     nextIndex        = index;
253            private int     expectedModCount = modCount;
254
255            @Override
256            public void add(E element) {
257                checkForConcurrentModification();
258                RankList.this.add(nextIndex++, element);
259                expectedModCount = modCount;
260                last             = null;
261            }
262
263            @Override
264            public boolean hasNext() {
265                return nextIndex < size();
266            }
267
268            @Override
269            public boolean hasPrevious() {
270                return nextIndex > 0;
271            }
272
273            @Override
274            public E next() {
275                checkForConcurrentModification();
276                if (!hasNext())
277                    throw new NoSuchElementException();
278                nextIndex++;
279                last = next;
280                next = next.next[0];
281                return last.element;
282            }
283
284            @Override
285            public int nextIndex() {
286                return nextIndex;
287            }
288
289            @Override
290            public E previous() {
291                checkForConcurrentModification();
292                if (!hasPrevious())
293                    throw new NoSuchElementException();
294                last = next = next.prev;
295                nextIndex--;
296                return next.element;
297            }
298
299            @Override
300            public int previousIndex() {
301                return nextIndex - 1;
302            }
303
304            @Override
305            public void remove() {
306                checkForConcurrentModification();
307                checkState(last != null);
308                if (last == next.prev)
309                    RankList.this.remove(nextIndex-- - 1);
310                else {
311                    next = next.next[0];
312                    RankList.this.remove(nextIndex);
313                }
314                expectedModCount = modCount;
315                last             = null;
316            }
317
318            @Override
319            public void set(E element) {
320                checkForConcurrentModification();
321                checkState(last != null);
322                last.element = element;
323            }
324
325            private void checkForConcurrentModification() {
326                if (expectedModCount != modCount)
327                    throw new ConcurrentModificationException();
328            }
329        };
330    }
331
332    @SuppressWarnings("unchecked")
333    @Override
334    public E remove(int index) {
335        checkElementIndex(index, size);
336        final Node<E>[] update = new Node[level];
337        Node<E>         node   = head;
338        int             pos    = 0;
339        for (int i = level - 1; i >= 0; i--) {
340            while (pos + node.dist[i] <= index) {
341                pos  += node.dist[i];
342                node  = node.next[i];
343            }
344            update[i] = node;
345        }
346        node = node.next[0];
347        delete(node, update);
348        return node.element;
349    }
350
351    @Override
352    public void clear() {
353        for (int i = 0; i < MAX_LEVEL; i++) {
354            head.next[i] = head;
355            head.dist[i] = 1;
356        }
357        head.prev = head;
358        modCount++;
359        size = 0;
360    }
361
362    @Override
363    public E set(int index, E element) {
364        checkElementIndex(index, size);
365        Node<E> node = search(index);
366        E       e    = node.element;
367        node.element = element;
368        return e;
369    }
370
371    /**
372     * Returns a shallow copy of this {@code RankList}. The elements themselves are not cloned.
373     * 
374     * @return a shallow copy of this list
375     */
376    @SuppressWarnings("unchecked")
377    @Override
378    public RankList<E> clone() {
379        RankList<E> clone;
380        try {
381            clone = (RankList<E>) super.clone();
382        } catch (final CloneNotSupportedException e) {
383            throw new InternalError();
384        }
385        for (int i = 0; i < MAX_LEVEL; i++) {
386            clone.head.next[i] = clone.head;
387            clone.head.dist[i] = 1;
388        }
389        clone.head.prev = clone.head;
390        clone.random    = ThreadLocalRandom.current();
391        clone.level     = 1;
392        clone.modCount  = 0;
393        clone.size      = 0;
394        clone.addAll(this);
395        return clone;
396    }

398    private void writeObject(final ObjectOutputStream oos) throws IOException {
399        oos.defaultWriteObject();
400        oos.writeInt(size);
401        for (E e : this)
402            oos.writeObject(e);
403    }
404
405    @SuppressWarnings("unchecked")
406    private void readObject(final ObjectInputStream ois) throws IOException, ClassNotFoundException {
407        ois.defaultReadObject();
408        head = new Node<E>(null, MAX_LEVEL);
409        for (int i = 0; i < MAX_LEVEL; i++) {
410            head.next[i] = head;
411            head.dist[i] = 1;
412        }
413        head.prev = head;
414        random    = ThreadLocalRandom.current();
415        level     = 1;
416        int size = ois.readInt();
417        for (int i = 0; i < size; i++)
418            add((E) ois.readObject());
419    }
420
421    // skip list
422
423    private static class Node<E> {
424        private E               element;
425        private Node<E>         prev;
426        private final Node<E>[] next;
427        private final int[]     dist;
428
429        @SuppressWarnings("unchecked")
430        private Node(final E element, final int size) {
431            this.element = element;
432            next         = new Node[size];
433            dist         = new int[size];
434        }
435    }
436
437    private Node<E> search(final int index) {
438        Node<E> curr = head;
439        int     pos  = -1;
440        for (int i = level - 1; i >= 0; i--)
441            while (pos + curr.dist[i] <= index) {
442                pos  = pos + curr.dist[i];
443                curr = curr.next[i];
444            }
445        return curr;
446    }
447
448    private void delete(final Node<E> node, final Node<E>[] update) {
449        node.next[0].prev = node.prev;
450        for (int i = 0; i < level; i++)
451            if (update[i].next[i] == node) {
452                update[i].next[i]  = node.next[i];
453                update[i].dist[i] += node.dist[i] - 1;
454            } else
455                update[i].dist[i]--;
456        while (head.next[level - 1] == head && level > 1)
457            level--;
458        modCount++;
459        size--;
460    }
461
462    private int randomLevel() {
463        int randomLevel = 1;
464        while (randomLevel < MAX_LEVEL - 1 && random.nextDouble() < P)
465            randomLevel++;
466        return randomLevel;
467    }
