040/**
041 * An optionally bounded priority {@link Queue} based on a modified
042 * <a target="_blank" href="http://en.wikipedia.org/wiki/Red-black_tree">Red-Black Tree</a>. The elements of this queue
043 * are sorted according to their <i>natural ordering</i>, or by an explicit {@link Comparator} provided at creation.
044 * Attempting to insert {@code null} elements will succeed if the {@code Comparator} supports {@code null} values.
045 * Inserting non-comparable elements will result in a {@code ClassCastException}. This queue uses the same general
046 * ordering rules as a {@link PriorityQueue PriorityQueue}. The first element (the head) of this queue is considered to
047 * be the <i>least</i> element with respect to the specified ordering. A comparator is used ({@link Ordering#natural()
048 * whether or not one is explicitly provided}) to perform all element comparisons. Two elements which are deemed equal
049 * by the comparator's {@code compare(E, E)} method have equal priority from the standpoint of this queue. Elements with
050 * equal priority are sorted according to their insertion order.
051 * <p>
052 * Besides the regular {@link #peek() peek()}, {@link #poll() poll()}, {@link #remove() remove()} operations specified
053 * in the {@code Queue} interface, this implementation provides additional {@link #peekLast() peekLast()},
054 * {@link #pollLast() pollLast()}, {@link #removeLast() removeLast()} methods to examine the elements at the tail of the
055 * queue.
056 * <p>
057 * If this queue is bounded and becomes full the {@code offer(E)} method behaves according to the following policy: if
058 * the element to be added has higher priority than the lowest priority element currently in the queue, the new element
059 * is added and the lowest priority element is removed; else the new element is rejected. The {@code add(E)} and
060 * {@code addAll(Collection)} operations will throw an {@code IllegalStateException} when the queue is full and a new
061 * element is rejected; as required by the contract of {@link Queue#add Queue.add(E)}.
062 * <p>
063 * If the maximum size of this queue is not specified it will have an intrinsic capacity of {@code Integer.MAX_VALUE}
064 * elements.
065 * <p>
066 * Bounded priority queues are useful when implementing <i>n-best</i> algorithms (e.g. finding the <i>best</i> <i>n</i>
067 * elements in an arbitrary collection).
068 * <p>
069 * The {@link #iterator() iterator()} and {@link #descendingIterator()} methods return <i>fail-fast</i> iterators which
070 * are guaranteed to traverse the elements of the queue in ascending and descending priority order, respectively.
071 * Attempts to modify the elements in this queue at any time after an iterator is created, in any way except through the
072 * iterator's own remove method, will result in a {@code ConcurrentModificationException}.
073 * <p>
074 * This queue is not <i>thread-safe</i>. If multiple threads modify this queue concurrently it must be synchronized
075 * externally.
076 * <p>
077 * The underlying Red-Black Tree provides the following running time compared to a {@link PriorityQueue PriorityQueue}
078 * (where <i>n</i> is the size of this queue and <i>m</i> is the size of the specified collection which is iterable in
079 * linear time):
080 * 
081 * <pre>
082 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
083 *   <tr>
084 *     <th style="text-align:center;" rowspan="2">Method</th>
085 *     <th style="text-align:center;" colspan="2">Running Time</th>
086 *   </tr>
087 *   <tr>
088 *     <td style="text-align:center;"><b>TreeQueue</b><br>(<i>amortized</i>)</td>
089 *     <td style="text-align:center;"><b>PriorityQueue</b><br>(<i>worst-case</i>)</td>
090 *   </tr>
091 *   <tr>
092 *     <td>{@link #addAll(Collection) addAll(Collection)}<br>{@link #containsAll(Collection) containsAll(Collection)}<br>{@link #retainAll(Collection) retainAll(Collection)}<br>{@link #removeAll(Collection) removeAll(Collection)}</td>
093 *     <td colspan="2" style="text-align:center;"><i>O(m log n)</i></td>
094 *   </tr>
095 *   <tr>
096 *     <td>{@link #add(Object) add(E)}<br>{@link #offer(Object) offer(E)}<br>{@link #remove(Object)}</td>
097 *     <td colspan="2" style="text-align:center;"><i>O(log n)</i></td>
098 *   </tr>
099 *   <tr>
100 *     <td>{@link #contains(Object)}</td>
101 *     <td bgcolor="FFCC99" style="text-align:center;"><i>O(log n)</i></td>
102 *     <td bgcolor="FFCCCC" rowspan="2" style="text-align:center;"><i>O(n)</i></td>
103 *   </tr>
104 *   <tr>
105 *     <td>{@link #clear()}</td>
106 *     <td bgcolor="FFCC99" rowspan="2" style="text-align:center;"><i>O(1)</i></td>
107 *   </tr>
108 *   <tr>
109 *     <td>{@link #poll()}<br>{@link #remove() remove()}<br></td>
110 *     <td bgcolor="FFCCCC" style="text-align:center;"><i>O(log n)</i></td>
111 *   </tr>
112 *   <tr>
113 *     <td>{@link #element() element()}<br>{@link #isEmpty() isEmpty()}<br>{@link #peek()}<br>{@link #size()}</td>
114 *     <td colspan="2" style="text-align:center;"><i>O(1)</i></td>
115 *   </tr>
116 *   <tr>
117 *     <td>{@link #peekLast() peekLast()}<br>{@link #pollLast() pollLast()}<br>{@link #removeLast() removeLast()}</td>

122 * </pre>

155    /**
156     * Creates a new unbounded {@code TreeQueue} containing the specified initial elements. If {@code elements} is an
157     * instance of {@link SortedSet} , {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or {@code SortedCollection} this
158     * queue will be ordered according to the same ordering. Otherwise, this queue will be ordered according to the
159     * <i>natural ordering</i> of its elements.
160     * 
161     * @param elements the collection whose elements are to be placed into the queue
162     * @return a new unbounded {@code TreeQueue} containing the elements of the specified collection
163     * @throws ClassCastException   if elements of the specified collection cannot be compared to one another according to
164     *                              this queue's ordering
165     * @throws NullPointerException if any of the elements of the specified collection or the collection itself is
166     *                              {@code null}
167     */
168    @SuppressWarnings({ "unchecked" })
169    public static <E> TreeQueue<E> create(final Iterable<? extends E> elements) {
170        checkNotNull(elements, "elements == null");
171
172        Comparator<? super E> comparator = null;
173        if (elements instanceof SortedSet<?>)
174            comparator = ((SortedSet<? super E>) elements).comparator();
175        else if (elements instanceof PriorityQueue<?>)
176            comparator = ((PriorityQueue<? super E>) elements).comparator();
177        else if (elements instanceof SortedCollection<?>)
178            comparator = ((SortedCollection<? super E>) elements).comparator();
179        else if (elements instanceof MinMaxPriorityQueue<?>)
180            comparator = ((MinMaxPriorityQueue<? super E>) elements).comparator();
181
182        if (comparator == null)
183            comparator = (Comparator<? super E>) Ordering.natural();
184
185        return orderedBy(comparator).create(elements);
186    }

340    @Override
341    public boolean offer(E e) {
342//        checkNotNull(e, "e == null");
343        if (size() == capacity)
344            if (comparator().compare(e, peekLast()) < 0)
345                pollLast();
346            else
347                return false;
348        final Node newNode = new Node(e);
349        insert(newNode);
350        return true;
351    }

391    /**
392     * Returns an iterator over the elements of this queue in priority order from first (head) to last (tail).
393     * 
394     * @return an iterator over the elements of this queue in priority order
395     */
396    @Override
397    public Iterator<E> iterator() {
398        return new Iterator<E>() {
399            private Node next = min;
400            private Node last = nil;
401            private int expectedModCount = modCount;
402
403            @Override
404            public boolean hasNext() {
405                return next != nil;
406            }
407
408            @Override
409            public void remove() {
410                checkForConcurrentModification();
411                if (last == nil)
412                    throw new IllegalStateException();
413                if (last.left != nil && last.right != nil)
414                    next = last;
415                delete(last);
416                expectedModCount = modCount;
417                last = nil;
418            }
419
420            @Override
421            public E next() {
422                checkForConcurrentModification();
423                if (next == nil)
424                    throw new NoSuchElementException();
425                last = next;
426                next = successor(next);
427                return last.element;
428            }
429
430            private void checkForConcurrentModification() {
431                if (modCount != expectedModCount)
432                    throw new ConcurrentModificationException();
433            }
434        };
435    }

455                checkForConcurrentModification();

468                    throw new NoSuchElementException();

475                if (modCount != expectedModCount)
476                    throw new ConcurrentModificationException();
477            }
478        };

532    private class Node {
533        private E element = null;
534        private Node parent, left, right;
535        private Color color = BLACK;
536
537        private Node() {
538            parent = this;
539            right = this;
540            left = this;
541        }

544            this.element = element;
545            parent = nil;
546            right = nil;
547            left = nil;

551    /**
552     * Introduction to Algorithms (CLR) Second Edition
553     * 
554     * <pre>
555     * RB-INSERT(T, z)
556     * y = nil[T]
557     * x = root[T]
558     * while x != nil[T]
559     *    do y = x
560     *       if key[z] < key[x]
561     *          then x = left[x]
562     *          else x = right[x]
563     * p[z] = y
564     * if y = nil[T]
565     *    then root[T] = z
566     *    else if key[z] < key[y]
567     *            then left[y] = z
568     *            else right[y] = z
569     * left[z] = nil[T]
570     * right[z] = nil[T]
571     * color[z] = RED
572     * RB-INSERT-FIXUP(T, z)
573     * </pre>
574     */
575    private boolean insert(final Node z) {
576        modCount++;
577        Node x = root;
578        Node y = nil;
579        while (x != nil) {
580            y = x;
581            int cmp = comparator.compare(z.element, x.element);
582            if (cmp == 0)
583                return false;
584            else if (cmp < 0)
585                x = x.left;
586            else
587                x = x.right;
588        }
589        z.parent = y;
590        if (y == nil)
591            root = z;
592        else if (comparator.compare(z.element, y.element) < 0)
593            y.left = z;
594        else
595            y.right = z;
596        fixAfterInsertion(z);
597        if (min == nil || comparator.compare(z.element, min.element) < 0)
598            min = z;
599        size++;
600        return true;
601    }

603    /**
604     * Introduction to Algorithms (CLR) Second Edition
605     * 
606     * <pre>
607     * RB-DELETE-FIXUP(T, z)
608     * if left[z] = nil[T] or right[z] = nil[T]
609     *    then y = z
610     *    else y = TREE-SUCCESSOR(z)
611     * if left[y] != nil[T]
612     *    then x = left[y]
613     *    else x = right[y]
614     * p[x] = p[y]
615     * if p[y] = nil[T]
616     *    then root[T] = x
617     *    else if y = left[p[y]]
618     *            then left[p[y]] = x
619     *            else right[p[y]] = x
620     * if y != z
621     *    then key[z] = key[y]
622     *         copy y's satellite data into z
623     * if color[y] = BLACK
624     *    then RB-DELETE-FIXUP(T, x)
625     * return y
626     * </pre>
627     */
628    private void delete(Node z) {
629        size--;
630        modCount++;
631        Node x, y;
632        if (min == z)
633            min = successor(z);
634        if (z.left == nil || z.right == nil)
635            y = z;
636        else
637            y = successor(z);
638        if (y.left != nil)
639            x = y.left;
640        else
641            x = y.right;
642        x.parent = y.parent;
643        if (y.parent == nil)
644            root = x;
645        else if (y == y.parent.left)
646            y.parent.left = x;
647        else
648            y.parent.right = x;
649        if (y != z)
650            z.element = y.element;
651        if (y.color == Color.BLACK)
652            fixAfterDeletion(x);
653    }

655    private Node search(final E e) {
656        Node n = root;
657        while (n != nil) {
658            int cmp = comparator.compare(e, n.element);
659            if (cmp == 0)
660                return n;
661            if (cmp < 0)
662                n = n.left;
663            else
664                n = n.right;
665        }
666        return null;
667    }

669    /**
670     * Introduction to Algorithms (CLR) Second Edition
671     * 
672     * <pre>
673     * TREE-SUCCESSOR(x)
674     * if right[x] != NIL
675     *    then return TREE-MINIMUM(right[x])
676     * y = p[x]
677     * while y != NIL and x = right[y]
678     *    do x = y
679     *       y = p[y]
680     * return y
681     * </pre>
682     */
683    private Node successor(Node x) {
684        if (x == nil)
685            return nil;
686        if (x.right != nil) {
687            Node y = x.right;
688            while (y.left != nil)
689                y = y.left;
690            return y;
691        }
692        Node y = x.parent;
693        while (y != nil && x == y.right) {
694            x = y;
695            y = y.parent;
696        }
697        return y;
698    }

700    private Node predecessor(Node x) {
701        if (x == nil)
702            return nil;
703        if (x.left != nil) {
704            Node y = x.left;
705            while (y.right != nil)
706                y = y.right;
707            return y;
708        }
709        Node y = x.parent;
710        while (y != nil && x == y.left) {
711            x = y;
712            y = y.left;
713        }
714        return y;
715    }

717    /**
718     * Introduction to Algorithms (CLR) Second Edition
719     * 
720     * <pre>
721     * LEFT-ROTATE(T, x)
722     * y = right[x]                         Set y.
723     * right[x] = left[y]                   Turn y's left subtree into x's right subtree.
724     * if left[y] != nil[T]
725     *    then p[left[y]] = x
726     * p[y] = p[x]                          Link x's parent to y.
727     * if p[x] = nil[T]
728     *    then root[T] = y
729     *    else if x = left[p[x]]
730     *            then left[p[x]] = y
731     *            else right[p[x]] = y
732     * left[y] = x                          Put x on y's left.
733     * p[x] = y
734     * </pre>
735     */
736    private void leftRotate(final Node x) {
737        if (x != nil) {
738            Node n = x.right;
739            x.right = n.left;
740            if (n.left != nil)
741                n.left.parent = x;
742            n.parent = x.parent;
743            if (x.parent == nil)
744                root = n;
745            else if (x.parent.left == x)
746                x.parent.left = n;
747            else
748                x.parent.right = n;
749            n.left = x;
750            x.parent = n;
751        }
752    }
753
754    private void rightRotate(final Node x) {
755        if (x != nil) {
756            Node n = x.left;
757            x.left = n.right;
758            if (n.right != nil)
759                n.right.parent = x;
760            n.parent = x.parent;
761            if (x.parent == nil)
762                root = n;
763            else if (x.parent.right == x)
764                x.parent.right = n;
765            else
766                x.parent.left = n;
767            n.right = x;
768            x.parent = n;
769        }
770    }

772    /**
773     * Introduction to Algorithms (CLR) Second Edition
774     * 
775     * <pre>
776     * RB-INSERT-FIXUP(T, z)
777     * while color[p[z]] = RED
778     *    do if p[z] = left[p[p[z]]]
779     *          then y = right[p[p[z]]]
780     *               if color[y] = RED
781     *                  then color[p[z]] = BLACK                    Case 1
782     *                       color[y] = BLACK                       Case 1 
783     *                       color[p[p[z]]] = RED                   Case 1
784     *                       z = p[p[z]]                            Case 1
785     *                  else if z = right[p[z]]
786     *                          then z = p[z]                       Case 2
787     *                               LEFT-ROTATE(T, z)              Case 2
788     *                       color[p[z]] = BLACK                    Case 3
789     *                       color[p[p[z]]] = RED                   Case 3
790     *                       RIGHT-ROTATE(T, p[p[z]])               Case 3
791     *          else (same as then clause
792     *                        with right and left exchanged)
793     * color[root[T]] = BLACK
794     * </pre>
795     */
796    private void fixAfterInsertion(Node z) {
797        z.color = RED;
798        while (z.parent.color == RED) {
799            if (z.parent == z.parent.parent.left) {
800                Node y = z.parent.parent.right;
801                if (y.color == RED) {
802                    z.parent.color = BLACK;
803                    y.color = BLACK;
804                    z.parent.parent.color = RED;
805                    z = z.parent.parent;
806                } else {
807                    if (z == z.parent.right) {
808                        z = z.parent;
809                        leftRotate(z);
810                    }
811                    z.parent.color = BLACK;
812                    z.parent.parent.color = RED;
813                    rightRotate(z.parent.parent);
814                }
815            } else {
816                Node y = z.parent.parent.left;
817                if (y.color == RED) {
818                    z.parent.color = BLACK;
819                    y.color = BLACK;
820                    z.parent.parent.color = RED;
821                    z = z.parent.parent;
822                } else {
823                    if (z == z.parent.left) {
824                        z = z.parent;
825                        rightRotate(z);
826                    }
827                    z.parent.color = BLACK;
828                    z.parent.parent.color = RED;
829                    leftRotate(z.parent.parent);
830                }
831            }
832        }
833        root.color = BLACK;
834    }

836    /**
837     * Introduction to Algorithms (CLR) Second Edition
838     * 
839     * <pre>
840     * RB-DELETE-FIXUP(T, x)
841     * while x != root[T] and color[x] = BLACK
842     *    do if x = left[p[x]]
843     *          then w = right[p[x]]
844     *               if color[w] = RED
845     *                  then color[w] = BLACK                               Case 1
846     *                       color[p[x]] = RED                              Case 1
847     *                       LEFT-ROTATE(T, p[x])                           Case 1
848     *                       w = right[p[x]]                                Case 1
849     *               if color[left[w]] = BLACK and color[right[w]] = BLACK
850     *                  then color[w] = RED                                 Case 2
851     *                       x = p[x]                                       Case 2
852     *                  else if color[right[w]] = BLACK
853     *                          then color[left[w]] = BLACK                 Case 3
854     *                               color[w] = RED                         Case 3
855     *                               RIGHT-ROTATE(T,w)                      Case 3
856     *                               w = right[p[x]]                        Case 3
857     *                       color[w] = color[p[x]]                         Case 4
858     *                       color[p[x]] = BLACK                            Case 4
859     *                       color[right[w]] = BLACK                        Case 4
860     *                       LEFT-ROTATE(T, p[x])                           Case 4
861     *                       x = root[T]                                    Case 4
862     *          else (same as then clause with right and left exchanged)
863     * color[x] = BLACK
864     * </pre>
865     */
866    private void fixAfterDeletion(Node x) {
867        while (x != root && x.color == BLACK) {
868            if (x == x.parent.left) {
869                Node w = x.parent.right;
870                if (w.color == RED) {
871                    w.color = BLACK;
872                    x.parent.color = RED;
873                    leftRotate(x.parent);
874                    w = x.parent.right;
875                }
876                if (w.left.color == BLACK && w.right.color == BLACK) {
877                    w.color = RED;
878                    x = x.parent;
879                } else {
880                    if (w.right.color == BLACK) {
881                        w.left.color = BLACK;
882                        w.color = RED;
883                        rightRotate(w);
884                        w = x.parent.right;
885                    }
886                    w.color = x.parent.color;
887                    x.parent.color = BLACK;
888                    w.right.color = BLACK;
889                    leftRotate(x.parent);
890                    x = root;
891                }
892            } else {
893                Node w = x.parent.left;
894                if (w.color == RED) {
895                    w.color = BLACK;
896                    x.parent.color = RED;
897                    rightRotate(w.parent);
898                    w = x.parent.left;
899                }
900                if (w.right.color == BLACK && w.left.color == BLACK) {
901                    w.color = RED;
902                    x = x.parent;
903                } else {
904                    if (w.left.color == BLACK) {
905                        w.right.color = BLACK;
906                        w.color = RED;
907                        leftRotate(w);
908                        w = x.parent.left;
909                    }
910                    w.color = x.parent.color;
911                    w.parent.color = BLACK;
912                    w.left.color = BLACK;
913                    rightRotate(x.parent);
914                    x = root;
915                }
916            }
917        }
918        x.color = BLACK;
919    }
