075    /**
076     * Closes the specified {@code ZipFile}, with control over whether an {@code IOException} may be thrown. If the ZIP file
077     * is {@code null} it is ignored. This is primarily useful in a finally block, where a thrown exception needs to be
078     * logged but not propagated to the caller (otherwise the original exception will be lost).
079     * <p>
080     * If an {@code IOException} occurs and {@code hideIOException} is {@code true} it will be logged but never thrown.
081     * Otherwise it will be propagated to the caller.
082     * 
083     * @deprecated Java 7+ users should switch to <a target="_blank" href=
084     *             "https://docs.oracle.com/javase/tutorial/essential/exceptions/tryResourceClose.html">try-with-resources</a>
085     *             or Guava's {@link Closeables#close(Closeable, boolean)} instead.
086     * 
087     * @param zip             the specified ZIP file
088     * @param hideIOException whether or not propagate {@code IOException}s
089     * @throws IOException if an I/O error occurs and {@code hideIOException} is {@code false}
090     */
091    public static void close(final ZipFile zip, final boolean hideIOException) throws IOException {
092        try {
093            if (zip != null)
094                zip.close();
095        } catch (final IOException e) {
096            if (hideIOException)
097                logger.log(WARNING, "An IOException occurred while attempting to close " + zip.getName(), e);
098            else
099                throw e;
100        }
101    }

103    /**
104     * Closes the specified {@code ZipFile}, logging and swallowing any {@code IOException}s. If the ZIP file is
105     * {@code null} it is ignored.
106     * <p>
107     * <b>Warning:</b> this method is prone to misuse. In general {@code IOException}s should be handled and/or propagated
108     * to the caller when closing a resource. Use this method only if you sure that no other action can be taken if an
109     * {@code IOException} occurs.
110     * 
111     * @deprecated Java 7+ users should switch to <a target="_blank" href=
112     *             "https://docs.oracle.com/javase/tutorial/essential/exceptions/tryResourceClose.html">try-with-resources</a>
113     *             or Guava's {@link Closeables#close(Closeable, boolean)} instead.
114     * 
115     * @param zip the specified ZIP file
116     */
117    public static void closeQuietly(final ZipFile zip) {
118        try {
119            close(zip, true);
120        } catch (final IOException e) {
121            throw new AssertionError();
122        }
123    }

270    /**
271     * Writes a new ZIP entry and positions the stream for writing the next entry.
272     * <p>
273     * An {@code IllegalArgumentException} will be thrown if the ZIP entry is a directory and the {@code bytes} array is not
274     * {@code null}.
275     * <p>
276     * <b>Note:</b> The directory structure of a ZIP file is inferred from the path of each constituent file. It is not
277     * necessary to create directory entries when writing a ZIP file unless you wish to create an empty directory.
278     * 
279     * @param zout  the {@code ZipOutputStream}
280     * @param ze    the {@code ZipEntry}
281     * @param bytes the contents of the {@code ZipEntry} or {@code null} if the {@code ZipEntry} is a directory
282     * @return the {@code ZipOutputStream}
283     * @throws IOException              if an I/O error occurs
284     * @throws IllegalArgumentException if the specified entry is a directory and the {@code bytes} array is not
285     *                                  {@code null}
286     */
287    public static ZipOutputStream putNextEntry(final ZipOutputStream zout, final ZipEntry ze, final byte[] bytes) throws IOException {
288        checkNotNull(zout, "zout == null");
289        checkNotNull(ze, "ze ==  null");
290        checkArgument(bytes == null || !ze.isDirectory(), "%s is a directory", ze);
291
292        zout.putNextEntry(ze);
293        if (bytes != null && bytes.length > 0)
294            zout.write(bytes);
295        zout.closeEntry();
296        return zout;
297    }

437    /**
438     * Extracts the given ZIP file to the destination directory using the given {@code Predicate} to filter ZIP entries and
439     * the specified charset to decode ZIP entry names. The charset is ignored if the
440     * <a target="_blank" href="package-summary.html#lang_encoding">language encoding bit</a> of the ZIP entry's general
441     * purpose bit flag is set.
442     * <p>
443     * Password protected ZIP files are not supported. If the destination directory does not exist this call will result in
444     * an {@code IllegalArgumentException}. If the ZIP file contains entries outside the destination directory a
445     * {@code ZipException} will be thrown. If the destination directory is a symbolic link it will be resolved to its final
446     * target. ZIP entries representing symbolic links are not supported. Existing files will be overwritten.
447     * <p>
448     * <b>Note:</b> If this operation fails because of an I/O error or other problems it may have succeeded in extracting
449     * some (but not all) of the files and/or directories.
450     * 
451     * @param zip     the given ZIP file
452     * @param filter  the predicate to use to filter ZIP entries
453     * @param dest    the destination directory
454     * @param charset the charset to use to decode the ZIP entry names (ignored if the language encoding bit of the ZIP
455     *                entry's general purpose bit flag is set)
456     * @return the destination directory
457     * @throws IOException  if an I/O error occurs
458     * @throws ZipException if a ZIP entry falls outside the destination directory (for example if the ZIP file is created
459     *                      using absolute pathnames)
460     */
461    public static Path unzip(final Path zip, final java.util.function.Predicate<ZipEntry> filter, final Path dest, final Charset charset) throws IOException {
462        checkNotNull(zip, "zip == null");
463        checkNotNull(dest, "dest == null");
464        checkNotNull(charset, "charset == null");
465        checkNotNull(filter, "filter == null");
466        checkArgument(Files.isDirectory(dest), "%s is not a directory or does not exist", dest);
467
468        final Path target = dest.toRealPath();
469
470        try (final ZipInputStream zis = newZipInputStream(zip, charset)) {
471
472            ZipEntry ze;
473            while ((ze = zis.getNextEntry()) != null)
474                if (filter.test(ze)) {
475                    final Path path = target.resolve(ze.getName());
476
477                    if (!path.startsWith(target))
478                        throw new ZipException(ze.getName() + " is outside the destination directory");
479
480                    if (ze.isDirectory())
481                        createDirectories(path);
482                    else
483                        Files.copy(zis, createParentDirectories(path), REPLACE_EXISTING);
484                }
485        }
486
487        return dest;
488    }
