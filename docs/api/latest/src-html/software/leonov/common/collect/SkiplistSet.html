038/**
039 * A {@code Set} implementation based on a modified <a href="http://en.wikipedia.org/wiki/Skip_list">skip list</a>.
040 * Elements are sorted from <i>least</i> to <i>greatest</i> according to their <i>natural ordering</i>, or by an
041 * explicit {@link Comparator} provided at creation. Attempting to insert {@code null} elements will succeed if the
042 * {@code Comparator} supports {@code null} values. Inserting non-comparable elements will result in a
043 * {@code ClassCastException}.
044 * <p>
045 * The iterators obtained from the {@link #iterator()} method are <i>fail-fast</i>. Attempts to modify the elements in
046 * this set at any time after an iterator is created, in any way except through the iterator's own remove method, will
047 * result in a {@code ConcurrentModificationException}.
048 * <p>
049 * This set is not <i>thread-safe</i>. If multiple threads modify this set concurrently it must be synchronized
050 * externally.
051 * <p>
052 * This implementation uses a comparator (whether or not one is explicitly provided) to perform all element comparisons.
053 * Two elements which are deemed equal by the comparator's {@code compare(E, E)} method are, from the standpoint of this
054 * set, equal.
055 * <p>
056 * Invented by <a href="http://www.cs.umd.edu/~pugh/">Bill Pugh</a> in 1990, A skip list is a probabilistic data
057 * structure for maintaining items in sorted order. Strictly speaking it is impossible to make any hard guarantees
058 * regarding the worst-case performance of this class. Practical performance is <i>expected</i> to be logarithmic with
059 * an extremely high degree of probability as the skip list grows.
060 * <p>
061 * The underlying array-based skip list provides the following expected case running time (where <i>n</i> is the size of
062 * this set and <i>m</i> is the size of the specified collection which is iterable in linear time):
063 * 
064 * <pre>
065 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
066 *   <tr>
067 *     <th style="text-align:center;">Method</th>
068 *     <th style="text-align:center;">Running Time</th>
069 *   </tr>
070 *   <tr>
071 *     <td>{@link #addAll(Collection) addAll(Collection)}<br/>{@link #containsAll(Collection) containsAll(Collection)}<br/>{@link #retainAll(Collection) retainAll(Collection)}<br/>{@link #removeAll(Collection) removeAll(Collection)}</td>
072 *     <td style="text-align:center;"><i>O(m log n)</i></td>
073 *   </tr>
074 *   <tr>
075 *     <td>{@link #add(Object) add(E)}<br/>{@link #contains(Object)}<br/>{@link #remove(Object)}<br/>{@link Iterator#remove()}</td>
076 *     <td style="text-align:center;"><i>O(log n)</i></td>
077 *   </tr>
078 *   <tr>
079 *     <td>{@link #clear() clear()}<br/>{@link #isEmpty() isEmpty()}<br/>{@link #size()}</td>
080 *     <td style="text-align:center;"><i>O(1)</i></td>
081 *   </tr>
082 * </table>
083 * </pre>
084 * 
085 * @deprecated <b>This class should only be used for testing. It provides no practical benefit when compared to
086 *             {@link java.util.TreeSet java.util.TreeSet} and lacks navigation methods defined in the
087 *             {@link NavigableSet} interface.</b>
088 * 
089 * @author Zhenya Leonov
090 * @param <E> the type of elements maintained by this set
091 * @see Skiplist
092 */
093final public class SkiplistSet<E> extends AbstractSet<E> implements SortedCollection<E>, Serializable, Cloneable {
094
095    private static final long serialVersionUID = 1L;
096    private static final double P = .5;
097    private static final int MAX_LEVEL = 32;
098    transient int size = 0;
099    private transient int level = 1;
100    private transient Random random = ThreadLocalRandom.current();
101    private transient Node<E> head = new Node<E>(null, MAX_LEVEL);
102    private final Comparator<? super E> comparator;
103    transient int modCount = 0;
104
105    private SkiplistSet(final Comparator<? super E> comparator) {
106        this.comparator = comparator;
107        for (int i = 0; i < MAX_LEVEL; i++)
108            head.next[i] = head;
109    }
110
111    /**
112     * Creates a new {@code SkiplistSet} that orders its elements according to their <i>natural ordering</i>.
113     * 
114     * @return a new {@code SkiplistSet} that orders its elements according to their <i>natural ordering</i>
115     */
116    // @SuppressWarnings("rawtypes")
117    // Use <T extends Comparable<?>> instead of the technically correct <T extends Comparable<? super T>> if using Java 6.
118    public static <E extends Comparable<? super E>> SkiplistSet<E> create() {
119        return new SkiplistSet<E>(Ordering.natural());
120    }
121
122    /**
123     * Creates a new {@code SkiplistSet} that uses the specified comparator to order its elements.
124     * 
125     * @param comparator the specified comparator
126     * @return a new {@code SkiplistSet} that uses the specified comparator to order its elements
127     * 
128     */
129    public static <E> SkiplistSet<E> create(final Comparator<? super E> comparator) {
130        return new SkiplistSet<E>(comparator);
131    }
132
133    /**
134     * Creates a new {@code SkiplistSet} containing the specified initial elements. If {@code elements} is an instance of
135     * {@link SortedSet}, {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or {@code SortedCollection}, this set will be
136     * ordered according to the same ordering. Otherwise, this set will be ordered according to the <i>natural ordering</i>
137     * of its elements.
138     * 
139     * @param elements the collection whose elements are to be placed into the set
140     * @return a new {@code SkiplistSet} containing the elements of the specified collection
141     * @throws ClassCastException   if elements of the specified collection cannot be compared to one another according to
142     *                              this set's ordering
143     * @throws NullPointerException if any of the elements of the specified collection or the collection itself is
144     *                              {@code null}
145     */
146    @SuppressWarnings({ "unchecked" })
147    public static <E> SkiplistSet<E> create(final Iterable<? extends E> elements) {
148        checkNotNull(elements, "elements == null");
149
150        Comparator<? super E> comparator = null;
151        if (elements instanceof SortedSet<?>)
152            comparator = ((SortedSet<? super E>) elements).comparator();
153        else if (elements instanceof PriorityQueue<?>)
154            comparator = ((PriorityQueue<? super E>) elements).comparator();
155        else if (elements instanceof SortedCollection<?>)
156            comparator = ((SortedCollection<? super E>) elements).comparator();
157        else if (elements instanceof MinMaxPriorityQueue<?>)
158            comparator = ((MinMaxPriorityQueue<? super E>) elements).comparator();
159
160        if (comparator == null)
161            comparator = (Comparator<? super E>) Ordering.natural();
162
163        final SkiplistSet<E> set = SkiplistSet.create(comparator);
164        Iterables.addAll(set, elements);
165        return set;
166    }
167
168    /**
169     * Returns the comparator used to order the elements in this set. If one was not explicitly provided a <i>natural
170     * order</i> comparator is returned.
171     * 
172     * @return the comparator used to order this set
173     */
174    @Override
175    public Comparator<? super E> comparator() {
176        return comparator;
177    }
178
179    /**
180     * Inserts the specified element into this set in sorted order.
181     */
182    @Override
183    public boolean add(E e) {
184//        checkNotNull(e, "e == null");
185        @SuppressWarnings("unchecked")
186        final Node<E>[] update = new Node[MAX_LEVEL];
187        final int newLevel = randomLevel();
188        Node<E> x = head;
189        Node<E> y = head;
190        int i;
191        for (i = level - 1; i >= 0; i--) {
192            while (x.next[i] != y && comparator.compare(x.next[i].element, e) < 0)
193                x = x.next[i];
194            y = x.next[i];
195            update[i] = x;
196        }
197        if (newLevel > level) {
198            for (i = level; i < newLevel; i++)
199                update[i] = head;
200            level = newLevel;
201        }
202        if (x.next() != this.head && comparator.compare(x.next().element, e) == 0)
203            return false;
204        x = new Node<E>(e, newLevel);
205        for (i = 0; i < level; i++) {
206            if (i < newLevel) {
207                x.next[i] = update[i].next[i];
208                update[i].next[i] = x;
209            }
210        }
211        modCount++;
212        size++;
213        return true;
214    }
215
216    @Override
217    @SuppressWarnings("unchecked")
218    public boolean contains(Object o) {
219//        return o != null && search((E) o) != null;
220        return search((E) o) != null;
221    }
222
223    @Override
224    public Iterator<E> iterator() {
225        return new Iterator<E>() {
226            private Node<E> node = head.next();
227            private Node<E> last = null;
228            private int index = 0;
229            private int expectedModCount = modCount;
230
231            @Override
232            public boolean hasNext() {
233                return index < size();
234            }
235
236            @Override
237            public E next() {
238                checkForConcurrentModification();
239                if (!hasNext())
240                    throw new NoSuchElementException();
241                index++;
242                last = node;
243                node = node.next();
244                return last.element;
245            }
246
247            @Override
248            public void remove() {
249                checkForConcurrentModification();
250                checkState(last != null);
251                // System.out.println("removing " + last.element);
252                SkiplistSet.this.remove(last);
253                index--;
254                expectedModCount = modCount;
255                last = null;
256            }
257
258            void checkForConcurrentModification() {
259                if (expectedModCount != modCount)
260                    throw new ConcurrentModificationException();
261            }
262        };
263    }
264
265    private boolean remove(final Node<E> node) {
266        @SuppressWarnings("unchecked")
267        final Node<E>[] update = new Node[MAX_LEVEL];
268        Node<E> curr = head;
269        for (int i = level - 1; i >= 0; i--) {
270            while (curr.next[i] != head && comparator.compare(curr.next[i].element, node.element) <= 0 && curr.next[i] != node)
271                curr = curr.next[i];
272            update[i] = curr;
273        }
274        curr = curr.next();
275        delete(curr, update);
276        return true;
277    }
278
279    @SuppressWarnings("unchecked")
280    @Override
281    public boolean remove(Object o) {
282//        if (o == null)
283//            return false;
284        final Node<E>[] update = new Node[MAX_LEVEL];
285        final E element = (E) o;
286        Node<E> curr = head;
287        for (int i = level - 1; i >= 0; i--) {
288            while (curr.next[i] != head && comparator.compare(curr.next[i].element, element) < 0)
289                curr = curr.next[i];
290            update[i] = curr;
291        }
292        curr = curr.next();
293        if (curr == head || comparator.compare(curr.element, element) != 0)
294            return false;
295        delete(curr, update);
296        return true;
297    }
298
299    @Override
300    public int size() {
301        return size;
302    }
303
304    @Override
305    public void clear() {
306        for (int i = 0; i < MAX_LEVEL; i++) {
307            head.next[i] = head;
308        }
309        modCount++;
310        size = 0;
311    }
312
313    /**
314     * Returns a shallow copy of this {@code SkiplistSet}. The elements themselves are not cloned.
315     * 
316     * @return a shallow copy of this skip set
317     */
318    @SuppressWarnings("unchecked")
319    @Override
320    public SkiplistSet<E> clone() throws CloneNotSupportedException {
321        SkiplistSet<E> clone;
322        try {
323            clone = (SkiplistSet<E>) super.clone();
324        } catch (java.lang.CloneNotSupportedException e) {
325            throw new InternalError();
326        }
327        for (int i = 0; i < MAX_LEVEL; i++) {
328            clone.head.next[i] = clone.head;
329        }
330        clone.random = ThreadLocalRandom.current();
331        clone.level = 1;
332        clone.modCount = 0;
333        clone.size = 0;
334        clone.addAll(this);
335        return clone;
336    }
337
338    private void writeObject(java.io.ObjectOutputStream oos) throws java.io.IOException {
339        oos.defaultWriteObject();
340        oos.writeInt(size);
341        for (E e : this)
342            oos.writeObject(e);
343    }
344
345    @SuppressWarnings("unchecked")
346    private void readObject(java.io.ObjectInputStream ois) throws java.io.IOException, ClassNotFoundException {
347        ois.defaultReadObject();
348        head = new Node<E>(null, MAX_LEVEL);
349        for (int i = 0; i < MAX_LEVEL; i++) {
350            head.next[i] = head;
351        }
352        random = ThreadLocalRandom.current();
353        level = 1;
354        int size = ois.readInt();
355        for (int i = 0; i < size; i++)
356            add((E) ois.readObject());
357    }
358
359    // skip list
360
361    private static class Node<E> {
362        private E element;
363        private final Node<E>[] next;
364
365        @SuppressWarnings("unchecked")
366        private Node(final E element, final int size) {
367            this.element = element;
368            next = new Node[size];
369        }
370
371        private Node<E> next() {
372            return next[0];
373        }
374    }
375
376    private int randomLevel() {
377        int randomLevel = 1;
378        while (randomLevel < MAX_LEVEL - 1 && random.nextDouble() < P)
379            randomLevel++;
380        return randomLevel;
381    }
382
383    private void delete(final Node<E> node, final Node<E>[] update) {
384        for (int i = 0; i < level; i++)
385            if (update[i].next[i] == node) {
386                update[i].next[i] = node.next[i];
387            }
388        while (head.next[level - 1] == head && level > 1)
389            level--;
390        modCount++;
391        size--;
392    }
393
394    private Node<E> search(final E element) {
395        Node<E> curr = head;
396        for (int i = level - 1; i >= 0; i--)
397            while (curr.next[i] != head && comparator.compare(curr.next[i].element, element) < 0)
398                curr = curr.next[i];
399        curr = curr.next();
400        if (curr != head && comparator.compare(curr.element, element) == 0)
401            return curr;
402        return null;
403    }
