086    /**
087     * Returns all possible string combinations of length {@code k} from the set of characters in {@code alphabet}.
088     * 
089     * @param alphabet the set of characters to use
090     * @param k        the length
091     * @return all possible string combinations of length {@code k} from the set of characters in {@code alphabet}
092     */
093    public static List<String> combinations(final CharSequence alphabet, int k) {
094        checkNotNull(alphabet, "alphabet == null");
095        checkArgument(k > 0, "k < 0");
096
097        final List<String> list = Lists.newArrayListWithCapacity((int) Math.pow(alphabet.length(), k));
098
099        if (k == 1)
100            return Lists.transform(Lists.charactersOf(alphabet), Functions.toStringFunction());
101        else {
102            final List<String> sublist = combinations(alphabet, k - 1);
103
104            for (int i = 0; i < alphabet.length(); i++)
105                for (int j = 0; j < sublist.size(); j++)
106                    list.add(alphabet.charAt(i) + sublist.get(j));
107
108            return list;
109        }
110    }

126    /**
127     * Returns the number of occurrences of {@code substr} in {@code str}.
128     * 
129     * @param str    the string to search in
130     * @param substr the string to search for
131     * @return the number of occurrences of {@code substr} in {@code str}
132     */
133    public static int count(final String str, final String substr) {
134        checkNotNull(str, "str == null");
135        checkNotNull(substr, "substr == null");
136
137        if (str.length() == 0 || substr.length() == 0)
138            return 0;
139
140        int count = 0;
141        int index = 0;
142
143        while ((index = str.indexOf(substr, index)) != -1) {
144            count++;
145            index += substr.length();
146        }
147
148        return count;
149    }

183    /**
184     * Returns a string with the indentation of each line in the specified character sequence adjusted.
185     * <p>
186     * If {@code count > 0} then the specified number of spaces (U+0020) are inserted at the start of each line.<br/>
187     * If {@code count < 0} then up to specified number of {@link CharMatcher#whitespace()} characters are removed from the
188     * start of each line.<br/>
189     * If {@code count == 0} then no changes are made.
190     * <p>
191     * If any changes are made all line terminators will be normalized to the {@code \n} (U+000A) character.
192     * <p>
193     * <b>Note:</b> This method has subtle differences from its Java 12 counterpart {@link String#indent(int)
194     * String.indent(int)}:
195     * <p>
196     * <ul>
197     * <li>This method does not normalize line terminators if {@code count} is 0, while {@code String.indent(int)} always
198     * normalizes all line terminators.</li>
199     * <li>This method uses {@link CharMatcher#whitespace()} to recognize whitespace characters, while
200     * {@code String.indent(int)} uses {@link Character#isWhitespace(int)}.</li>
201     * <li>This method does not append the {@code \n} (U+000A) character to the last line in the character sequence, while
202     * {@code String.indent(int)} does.</li>
203     * </ul>
204     * 
205     * @param chars the specified character sequence
206     * @param count number of leading whitespace characters to add {@code (n > 0)} or remove {@code (n < 0)}
207     * @return a string with adjusted indentation of every line
208     */
209    public static String indent(final CharSequence chars, final int count) {
210        checkNotNull(chars, "chars == null");
211        if (chars.length() == 0)
212            return "";
213
214        if (count == 0)
215            return chars.toString();
216
217        try {
218            Stream<String> stream = CharSource.wrap(chars).openBufferedStream().lines();
219
220            if (count > 0) {
221                final String spaces = Strings.repeat(" ", count);
222                stream = stream.map(s -> spaces + s);
223            } else if (count < 0)
224                stream = stream.map(s -> s.substring(Math.min(-count, NON_WHITESPACE.indexIn(s))));
225
226            final char                               lastChar = chars.charAt(chars.length() - 1);
227            final Collector<CharSequence, ?, String> joiner   = (lastChar == '\n' || lastChar == '\r' ? Collectors.joining("\n", "", "\n") : Collectors.joining("\n"));
228
229            return stream.collect(joiner);
230        } catch (IOException e) {
231            throw new AssertionError(); // cannot happen
232        }
233    }

249    /**
250     * Returns the index within {@code str} of the first occurrence of {@code substr}, starting at the specified index and
251     * ignoring case differences.
252     * <p>
253     * <b>Note:</b> This method uses the rules of the default locale.
254     *
255     * @param str       the specified string
256     * @param substr    the string to search for
257     * @param fromIndex the index from which to start the search
258     * @return the index within {@code str} of the first occurrence of {@code substr}, ignoring case differences
259     * @throws IllegalArgumentException        if {@code fromIndex} < 0
260     * @throws StringIndexOutOfBoundsException if {@code fromIndex} > the length of {@code str}
261     * @throws NullPointerException            if either {@code str} or {@code substr} is {@code null}
262     */
263    public static int indexOfIgnoreCase(final String str, final String substr, final int fromIndex) {
264        checkNotNull(str, "str == null");
265        checkNotNull(substr, "substr == null");
266
267        if (fromIndex < 0)
268            throw new IllegalArgumentException("fromIndex < 0");
269        if (fromIndex > str.length())
270            throw new StringIndexOutOfBoundsException(fromIndex);
271
272        if (substr.isEmpty())
273            return fromIndex;
274
275        final int limit = str.length() - substr.length() + 1;
276
277        if (fromIndex <= limit)
278            for (int i = fromIndex; i < limit; i++)
279                if (str.regionMatches(true, i, substr, 0, substr.length()))
280                    return i;
281
282        return -1;
283
284    }

452        checkNotNull(str, "str == null");
453        str = escapeEOLCharacters(str, CR_REPLACEMENT, LF_REPLACEMENT);
454        str = WHITESPACE_NOT_TAB.replaceFrom(str, SPACE_REPLACEMENT);

665    /**
666     * Returns the given character sequence truncated to the specified maximum length.
667     * <p>
668     * If the length of the given sequence is greater than {@code maxLength}, the returned string will be exactly
669     * {@code maxLength} in length and will end with {@code suffix} (if it is not {@code null}). Otherwise the sequence will
670     * be returned as a {@code String} with no changes.
671     * <p>
672     * This method is {@code null} safe.
673     * 
674     * @param chars     the given character sequence or {@code null}
675     * @param maxLength the specified maximum length
676     * @param suffix    the string to append to the truncated sequence or {@code null} for no suffix
677     * @return the given character sequence truncated to the specified maximum length or {@code null}
678     * @throws IllegalArgumentException if {@code maxLength} < 1
679     * @throws IllegalArgumentException if {@code suffix} is not {@code null} and its length is not less than
680     *                                  {@code maxLength}
681     */
682    public static String truncate(final CharSequence chars, final int maxLength, final String suffix) {
683        checkArgument(maxLength > 0, "maxLength < 1");
684
685        if (chars == null)
686            return null;
687
688        if (chars.length() <= maxLength)
689            return chars.toString();
690
691        checkArgument(suffix == null || maxLength > suffix.length(), "maxLength <= suffix.length()");
692
693        final StringBuilder sb = new StringBuilder(maxLength).append(chars, 0, suffix == null ? maxLength : maxLength - suffix.length());
694
695        if (suffix != null)
696            sb.append(suffix);
697
698        return sb.toString();
699    }

701    /**
702     * Returns {@code null} if the given character sequence is {@code null}, empty, or contains only
703     * {@link CharMatcher#whitespace() whitespace} characters, otherwise the original character sequence is returned.
704     * <p>
705     * This method is {@code null} safe.
706     *
707     * @param <T>   the type of CharSequence (typically {@code CharSequence} or a {@code String})
708     * @param chars the specified character sequence or {@code null}
709     * @return {@code null} if the given character sequence is {@code null}, empty, or contains only
710     *         {@link CharMatcher#whitespace() whitespace} characters
711     */
712    public static <T extends CharSequence> T whitespaceToNull(final T chars) {
713        if (isNullOrWhitespace(chars))
714            return null;
715        return chars;
716    }

718    private static String replace(final String str, final String substr, final String replacement, final boolean ignoreCase) {
719        int index = ignoreCase ? indexOfIgnoreCase(str, substr) : str.indexOf(substr);
720
721        if (index < 0)
722            return str;
723
724        final int s_length = str.length();
725        final int t_length = substr.length();
726
727        final StringBuilder sb = new StringBuilder(s_length);
728
729        int start = 0;
730        while (index >= 0) {
731            sb.append(str, start, index).append(replacement);
732            start = index + t_length;
733            index = ignoreCase ? indexOfIgnoreCase(str, substr, start) : str.indexOf(substr, start);
734        }
735
736        return sb.append(str, start, s_length).toString();
737    }
