199    /**
200     * A convenience method which adapts an iterator to the {@code Iterable} interface suitable for using in Java 5+
201     * <i>for-each</i> loops.
202     * <p>
203     * <b>Warning:</b> Although not explicitly stated in the documentation it is generally assumed that an {@code Iterable}
204     * object is able to produce multiple independent {@code Iterator}s. Thus the returned {@code Iterable} does not behave
205     * like a general purpose {@code Iterable} because it supports only a single call to {@link Iterable#iterator()}.
206     * Invoking the {@code iterator()} method to obtain subsequent iterators will result in an
207     * {@code IllegalStateException}.
208     * 
209     * @deprecated Jave 8+ users can create an {@code Iterable} from an {@code Iterator} using Java's lambda facility, for
210     *             example:
211     * 
212     *             <pre>
213     * for (final String s : (Iterable&lt;String&gt;)() -> iterator) {
214     *    ...
215     * }
216     *             </pre>
217     * 
218     * @param <T>      the element type
219     * @param iterator the underlying iterator
220     * @return an {@code Iterable} which returns the underlying iterator on the first call to {@link Iterable#iterator()}
221     */
222    public static <T> Iterable<T> in(final Iterator<T> iterator) {
223        checkNotNull(iterator, "iterator == null");
224
225        return new Iterable<T>() {
226            private boolean first = true;
227
228            @Override
229            public Iterator<T> iterator() {
230                if (!first)
231                    throw new IllegalStateException();
232                first = false;
233                return iterator;
234            }
235
236            @Override
237            public String toString() {
238                return Iterators.toString(iterator);
239            }
240        };
241    }

290    /**
291     * Returns a random sample of {@code k} elements from the given {@code Iterator} using
292     * <a target="_blank" href="https://en.wikipedia.org/wiki/Reservoir_sampling">reservoir sampling</a>.
293     * <p>
294     * If {@code k} exceeds the size of the given iterator the returned list will contain all of its elements.
295     * 
296     * @param <T>        the element type
297     * @param population the given {@code Iterator}
298     * @param k          the number of random items to chose
299     * @param random     the {@code Random} number generator
300     * @return a random sample of {@code k} elements from the given {@code Iterator}
301     */
302    public static <T> List<T> sample(final Iterator<? extends T> population, final int k, final Random random) {
303        checkNotNull(population, "population == null");
304        checkNotNull(random, "random == null");
305        checkArgument(k > 0, "k < 1");
306
307        final List<T> items = Lists.newArrayListWithCapacity(k);
308        int           count = 0;
309
310        while (population.hasNext()) {
311            final T item = population.next();
312
313            count++;
314            if (count <= k)
315                items.add(item);
316            else {
317                int r = random.nextInt(count);
318                if (r < k)
319                    items.set(r, item);
320            }
321        }
322
323        return items;
324    }

326    /**
327     * Returns a {@code ListIterator} that applies the specified {@code Function} to each element of the given list
328     * iterator.
329     * <p>
330     * The returned list iterator does not support {@code add(E)} and {@code set(E)} but does support {@code remove()} if
331     * the given list iterator does.
332     * 
333     * @param <E>      the element type traversed by the given list iterator
334     * @param <T>      the type of output of the specified function
335     * @param from     the given list iterator
336     * @param function the specified function
337     * @return a list iterator that applies {@code function} to each element of {@code fromIterator}
338     */
339    public static <E, T> ListIterator<T> transform(final ListIterator<E> from, final Function<? super E, ? extends T> function) {
340        checkNotNull(from, "from == null");
341        checkNotNull(function, "function == null");
342
343        return new ForwardingListIterator<T>() {
344            @Override
345            public void add(T e) {
346                throw new UnsupportedOperationException();
347            }
348
349            @SuppressWarnings("unchecked")
350            @Override
351            // ok since type T is only used in next, previous, add, and set
352            protected ListIterator<T> delegate() {
353                return (ListIterator<T>) from;
354            }
355
356            @Override
357            public T next() {
358                return function.apply(from.next());
359            }
360
361            @Override
362            public T previous() {
363                return function.apply(from.previous());
364            }
365
366            @Override
367            public void set(T e) {
368                throw new UnsupportedOperationException();
369            }
370        };
371    }

447     * Returns an {@link Iterables#unmodifiableIterable unmodifiable} iterable which combines a single element with the
448     * specified iterable. The returned iterable produces iterators that lazily traverse the {@code first} element followed
449     * by the {@code rest} of the elements.
450     * 
451     * @param <E>   the element type
452     * @param first the first element
453     * @param rest  the rest of the elements
454     * @return an iterable which combines a single element with the specified iterable
455     */
456    public static <E> Iterable<E> concat(final E first, final Iterable<? extends E> rest) {
457        checkNotNull(rest, "rest == null");
458        return Iterables.unmodifiableIterable(Iterables.concat(Collections.singleton(first), rest));
459    }
