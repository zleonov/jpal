050/**
051 * A {@link Sortedlist} implementation, based on a modified
052 * <a target="_blank" href="http://en.wikipedia.org/wiki/Red-black_tree">Red-Black Tree</a>. Elements are sorted from
053 * <i>least</i> to <i>greatest</i> according to their <i>natural ordering</i>, or by an explicit {@link Comparator}
054 * provided at creation. Attempting to insert {@code null} elements will succeed if the {@code Comparator} supports
055 * {@code null} values. Inserting non-comparable elements will result in a {@code ClassCastException}.
056 * <p>
057 * The iterators obtained from the {@link #iterator()} and {@link #listIterator()} methods are <i>fail-fast</i>.
058 * Attempts to modify the elements in this sorted-list at any time after an iterator is created, in any way except
059 * through the iterator's own remove method, will result in a {@code ConcurrentModificationException}. Further, the list
060 * iterator does not support the {@code add(E)} and {@code set(E)} operations.
061 * <p>
062 * This sorted-list is not <i>thread-safe</i>. If multiple threads modify this sorted-list concurrently it must be
063 * synchronized externally.
064 * <p>
065 * This implementation uses a comparator ({@link Ordering#natural() whether or not one is explicitly provided}) to
066 * perform all element comparisons. Two elements which are deemed equal by the comparator's {@code compare(E, E)} method
067 * are, from the standpoint of this sorted-list, equal. Further, no guarantee is made as to the final order of
068 * <i>equal</i> elements. Ties may be broken arbitrarily.
069 * <p>
070 * The underlying Red-Black Tree provides the following amortized running time (where <i>n</i> is the size of this
071 * sorted-list and <i>m</i> is the size of the specified collection which is iterable in linear time):
072 * 
073 * <pre>
074 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
075 *   <tr>
076 *     <th style="text-align:center;">Method</th>
077 *     <th style="text-align:center;">Running Time</th>
078 *   </tr>
079 *   <tr>
080 *     <td>{@link #addAll(Collection) addAll(Collection)}<br>{@link #containsAll(Collection) containsAll(Collection)}<br>{@link #retainAll(Collection) retainAll(Collection)}<br>{@link #removeAll(Collection) removeAll(Collection)}</td>
081 *     <td style="text-align:center;"><i>O(m log n)</i></td>
082 *   </tr>
083 *   <tr>
084 *     <td>{@link #indexOf(Object)}<br>{@link #lastIndexOf(Object)}<br>{@link #get(int)}<br>{@link #remove(int)}<br>{@link #listIterator(int)}</td>
085 *     <td style="text-align:center;"><i>O(n)</i></td>
086 *   </tr>
087 *   <tr>
088 *     <td>{@link #add(Object) add(E)}<br>{@link #contains(Object)}<br>{@link #remove(Object)}</td>
089 *     <td style="text-align:center;"><i>O(log n)</i></td>
090 *   </tr>
091 *   <tr>
092 *     <td>{@link #clear() clear()}<br>{@link #isEmpty() isEmpty()}<br>{@link #size()}<br>{@link Iterator#remove()}<br>{@link ListIterator#remove()}</td>
093 *     <td style="text-align:center;"><i>O(1)</i></td>
094 *   </tr>
095 * </table>
096 * </pre>
097 * 
098 * The {@code subList} views exhibit identical time complexity, with the exception of the {@code clear()} operation
099 * which runs in linear time proportional to the size of the view.
100 * 
101 * @author Zhenya Leonov
102 * @param <E> the type of elements maintained by this sorted-list
103 * @see Skiplist
104 */
105public class Treelist<E> extends AbstractCollection<E> implements Sortedlist<E>, Cloneable, Serializable {

132    /**
133     * Creates a new {@code Treelist} containing the specified initial elements. If {@code elements} is an instance of
134     * {@link SortedSet}, {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or {@code SortedCollection}, this sorted-list
135     * will be ordered according to the same ordering. Otherwise, this sorted-list will be ordered according to the
136     * <i>natural ordering</i> of its elements.
137     * 
138     * @param elements the collection whose elements are to be placed into the sorted-list
139     * @return a new {@code Treelist} containing the elements of the specified collection
140     * @throws ClassCastException   if elements of the specified collection cannot be compared to one another according to
141     *                              this sorted-list's ordering
142     * @throws NullPointerException if any of the elements of the specified collection or the collection itself is
143     *                              {@code null}
144     */
145    @SuppressWarnings({ "unchecked" })
146    public static <E> Treelist<E> create(final Iterable<? extends E> elements) {
147        checkNotNull(elements, "elements == null");
148
149        Comparator<? super E> comparator = null;
150        if (elements instanceof SortedSet<?>)
151            comparator = ((SortedSet<? super E>) elements).comparator();
152        else if (elements instanceof PriorityQueue<?>)
153            comparator = ((PriorityQueue<? super E>) elements).comparator();
154        else if (elements instanceof SortedCollection<?>)
155            comparator = ((SortedCollection<? super E>) elements).comparator();
156        else if (elements instanceof MinMaxPriorityQueue<?>)
157            comparator = ((MinMaxPriorityQueue<? super E>) elements).comparator();
158
159        if (comparator == null)
160            comparator = (Comparator<? super E>) Ordering.natural();
161
162        return orderedBy(comparator).create(elements);
163    }

266    @Override
267    public int indexOf(Object o) {
268        if (o != null) {
269            @SuppressWarnings("unchecked")
270            final E e = (E) o;
271            final ListIterator<E> itor = listIterator();
272            while (itor.hasNext())
273                if (comparator.compare(itor.next(), e) == 0)
274                    return itor.previousIndex();
275        }
276        return -1;
277    }

279    @Override
280    public int lastIndexOf(Object o) {
281        if (o != null) {
282            @SuppressWarnings("unchecked")
283            final E e = (E) o;
284            final ListIterator<E> itor = listIterator();
285            while (itor.hasNext())
286                if (comparator.compare(itor.next(), e) == 0) {
287                    while (itor.hasNext())
288                        if (comparator.compare(itor.next(), e) != 0) {
289                            itor.previous();
290                            break;
291                        }
292                    return itor.previousIndex();
293                }
294        }
295        return -1;
296    }

303    @Override
304    public ListIterator<E> listIterator() {
305        return new ListIterator<E>() {
306            private int index = 0;
307            private Node next = min;
308            private Node prev = nil;
309            private Node last = nil;
310            private int expectedModCount = modCount;
311
312            @Override
313            public void add(E e) {
314                throw new UnsupportedOperationException();
315            }
316
317            @Override
318            public boolean hasNext() {
319                return index < size();
320            }
321
322            @Override
323            public boolean hasPrevious() {
324                return index > 0;
325            }
326
327            @Override
328            public E next() {
329                checkForConcurrentModification();
330                if (index == size())
331                    throw new NoSuchElementException();
332                prev = next;
333                index++;
334                next = successor(prev);
335                last = prev;
336                return prev.element;
337            }
338
339            @Override
340            public int nextIndex() {
341                return index;
342            }
343
344            @Override
345            public E previous() {
346                checkForConcurrentModification();
347                if (index == 0)
348                    throw new NoSuchElementException();
349                next = prev;
350                index--;
351                prev = predecessor(next);
352                last = next;
353                return next.element;
354            }
355
356            @Override
357            public int previousIndex() {
358                return index - 1;
359            }
360
361            @Override
362            public void remove() {
363                checkForConcurrentModification();
364                checkState(last != nil);
365                if (last.left != nil && last.right != nil)
366                    next = last;
367                delete(last);
368                index--;
369                expectedModCount = modCount;
370                last = nil;
371            }
372
373            @Override
374            public void set(E e) {
375                throw new UnsupportedOperationException();
376            }
377
378            private void checkForConcurrentModification() {
379                if (expectedModCount != modCount)
380                    throw new ConcurrentModificationException();
381            }
382        };
383    }

437    @Override
438    public boolean equals(Object o) {
439        if (o == this)
440            return true;
441        if (!(o instanceof Sortedlist))
442            return false;
443        try {
444            @SuppressWarnings("unchecked")
445            final Iterator<E> i = ((Collection<E>) o).iterator();
446            for (E e : this)
447                if (comparator.compare(e, i.next()) != 0)
448                    return false;
449            return !i.hasNext();
450        } catch (final ClassCastException e) {
451            return false;
452        } catch (final NullPointerException e) {
453            return false;
454        }
455    }

507    @SuppressWarnings("serial")
508    private class Sublist extends Treelist<E> {
509        private final Treelist<E> list;
510        private final int offset;
511        private Node from;
512        private Node to;
513
514        private void checkForConcurrentModification() {
515            if (this.modCount != list.modCount)
516                throw new ConcurrentModificationException();
517        }
518
519        public Sublist(final Treelist<E> list, int fromIndex, int toIndex) {
520            super(list.comparator);
521            this.list = list;
522            from = list.min;
523            offset = fromIndex;
524            this.modCount = list.modCount;
525            this.size = toIndex - fromIndex;
526            int i = 0;
527            for (; i < fromIndex; i++)
528                from = successor(from);
529            to = from;
530            for (; i < toIndex - 1; i++)
531                to = successor(to);
532        }
533
534        @Override
535        public boolean add(E e) {
536            checkForConcurrentModification();
537            if (comparator.compare(e, from.element) < 0 || comparator.compare(e, to.element) > 0)
538                throw new IllegalArgumentException("element out of range");
539            list.add(e);
540            this.modCount = list.modCount;
541            this.size++;
542            if (comparator.compare(to.element, e) <= 0)
543                to = successor(to);
544            return true;
545        }
546
547        @SuppressWarnings("unchecked")
548        @Override
549        public boolean contains(Object o) {
550            checkForConcurrentModification();
551            return o != null && search((E) o) != null;
552        }
553
554        @Override
555        public E get(int index) {
556            checkForConcurrentModification();
557            checkElementIndex(index, this.size);
558            return list.get(index + offset);
559        }
560
561        @Override
562        public ListIterator<E> listIterator() {
563            return listIterator(0);
564        }
565
566        @Override
567        public ListIterator<E> listIterator(final int index) {
568            checkForConcurrentModification();
569            checkPositionIndex(index, Sublist.this.size);
570            return new ListIterator<E>() {
571                private ListIterator<E> i = list.listIterator(index + offset);
572
573                @Override
574                public boolean hasNext() {
575                    return nextIndex() < Sublist.this.size;
576                }
577
578                @Override
579                public E next() {
580                    if (hasNext())
581                        return i.next();
582                    else
583                        throw new NoSuchElementException();
584                }
585
586                @Override
587                public boolean hasPrevious() {
588                    return previousIndex() >= 0;
589                }
590
591                @Override
592                public E previous() {
593                    if (hasPrevious())
594                        return i.previous();
595                    else
596                        throw new NoSuchElementException();
597                }
598
599                @Override
600                public int nextIndex() {
601                    return i.nextIndex() - offset;
602                }
603
604                @Override
605                public int previousIndex() {
606                    return i.previousIndex() - offset;
607                }
608
609                @Override
610                public void remove() {
611                    i.remove();
612                    Sublist.this.modCount = list.modCount;
613                    Sublist.this.size--;
614                }
615
616                @Override
617                public void set(E e) {
618                    throw new UnsupportedOperationException();
619                }
620
621                @Override
622                public void add(E e) {
623                    throw new UnsupportedOperationException();
624                }
625            };
626        }
627
628        @SuppressWarnings("unchecked")
629        @Override
630        public boolean remove(Object o) {
631            checkForConcurrentModification();
632            checkNotNull(o, "o == null");
633            final Node node = search((E) o);
634            if (node == null)
635                return false;
636            if (node == to)
637                to = predecessor(to);
638            if (node == from)
639                from = successor(from);
640            list.delete(node);
641            this.modCount = list.modCount;
642            this.size--;
643            return true;
644        }
645
646        @Override
647        public E remove(int index) {
648            checkForConcurrentModification();
649            checkElementIndex(index, this.size);
650            if (index == 0)
651                from = successor(from);
652            if (index == this.size - 1)
653                to = predecessor(to);
654            final E e = list.remove(index + offset);
655            this.modCount = list.modCount;
656            this.size--;
657            return e;
658        }
659
660        @Override
661        public int size() {
662            checkForConcurrentModification();
663            return this.size;
664        }
665
666        @Override
667        public void clear() {
668            checkForConcurrentModification();
669            final Iterator<E> iterator = iterator();
670            while (iterator.hasNext()) {
671                iterator.next();
672                iterator.remove();
673            }
674        }
675
676        @Override
677        public Treelist<E> clone() throws CloneNotSupportedException {
678            throw new CloneNotSupportedException();
679        }
680
681        private void writeObject(final ObjectOutputStream oos) throws NotSerializableException {
682            throw new NotSerializableException();
683        }
684
685        private void readObject(final ObjectInputStream ois) throws NotSerializableException {
686            throw new NotSerializableException();
687        }
688
689        @Override
690        Node search(final E e) {
691            final int compareFrom = comparator.compare(e, from.element);
692            final int compareTo = comparator.compare(e, to.element);
693            if (compareFrom < 0 || compareTo > 0)
694                return null;
695            if (compareFrom == 0)
696                return from;
697            else if (compareTo == 0)
698                return to;
699            else
700                return list.search(e);
701        }
702    }

791    private class Node {
792        private E element = null;
793        private Node parent, left, right;
794        private Color color = BLACK;
795
796        private Node() {
797            parent = this;
798            right = this;
799            left = this;
800        }

802        private Node(final E element) {
803            this.element = element;
804            parent = nil;
805            right = nil;
806            left = nil;
807        }

810    Node search(final E e) {
811        Node n = root;
812        while (n != nil) {
813            int cmp = comparator.compare(e, n.element);
814            if (cmp == 0)
815                return n;
816            if (cmp < 0)
817                n = n.left;
818            else
819                n = n.right;
820        }
821        return null;
822    }

824    /**
825     * Introduction to Algorithms (CLR) Second Edition
826     * 
827     * <pre>
828     * RB-INSERT(T, z)
829     * y = nil[T]
830     * x = root[T]
831     * while x != nil[T]
832     *    do y = x
833     *       if key[z] < key[x]
834     *          then x = left[x]
835     *          else x = right[x]
836     * p[z] = y
837     * if y = nil[T]
838     *    then root[T] = z
839     *    else if key[z] < key[y]
840     *            then left[y] = z
841     *            else right[y] = z
842     * left[z] = nil[T]
843     * right[z] = nil[T]
844     * color[z] = RED
845     * RB-INSERT-FIXUP(T, z)
846     * </pre>
847     */
848    private void insert(Node z) {
849        size++;
850        modCount++;
851        Node x = root;
852        Node y = nil;
853        while (x != nil) {
854            y = x;
855            if (comparator.compare(z.element, x.element) < 0)
856                x = x.left;
857            else
858                x = x.right;
859        }
860        z.parent = y;
861        if (y == nil)
862            root = z;
863        else if (comparator.compare(z.element, y.element) < 0)
864            y.left = z;
865        else
866            y.right = z;
867        fixAfterInsertion(z);
868        if (max == nil || comparator.compare(z.element, max.element) >= 0)
869            max = z;
870        if (min == nil || comparator.compare(z.element, min.element) < 0)
871            min = z;
872    }

874    /**
875     * Introduction to Algorithms (CLR) Second Edition
876     * 
877     * <pre>
878     * RB-DELETE-FIXUP(T, z)
879     * if left[z] = nil[T] or right[z] = nil[T]
880     *    then y = z
881     *    else y = TREE-SUCCESSOR(z)
882     * if left[y] != nil[T]
883     *    then x = left[y]
884     *    else x = right[y]
885     * p[x] = p[y]
886     * if p[y] = nil[T]
887     *    then root[T] = x
888     *    else if y = left[p[y]]
889     *            then left[p[y]] = x
890     *            else right[p[y]] = x
891     * if y != z
892     *    then key[z] = key[y]
893     *         copy y's satellite data into z
894     * if color[y] = BLACK
895     *    then RB-DELETE-FIXUP(T, x)
896     * return y
897     * </pre>
898     */
899    private void delete(Node z) {
900        size--;
901        modCount++;
902        Node x, y;
903        if (min == z)
904            min = successor(z);
905        if (max == z)
906            max = predecessor(z);
907        if (z.left == nil || z.right == nil)
908            y = z;
909        else
910            y = successor(z);
911        if (y.left != nil)
912            x = y.left;
913        else
914            x = y.right;
915        x.parent = y.parent;
916        if (y.parent == nil)
917            root = x;
918        else if (y == y.parent.left)
919            y.parent.left = x;
920        else
921            y.parent.right = x;
922        if (y != z)
923            z.element = y.element;
924        if (y.color == Color.BLACK)
925            fixAfterDeletion(x);
926    }

928    /**
929     * Introduction to Algorithms (CLR) Second Edition
930     * 
931     * <pre>
932     * TREE-SUCCESSOR(x)
933     * if right[x] != NIL
934     *    then return TREE-MINIMUM(right[x])
935     * y = p[x]
936     * while y != NIL and x = right[y]
937     *    do x = y
938     *       y = p[y]
939     * return y
940     * </pre>
941     */
942    private Node successor(Node x) {
943        if (x == nil)
944            return nil;
945        if (x.right != nil) {
946            Node y = x.right;
947            while (y.left != nil)
948                y = y.left;
949            return y;
950        }
951        Node y = x.parent;
952        while (y != nil && x == y.right) {
953            x = y;
954            y = y.parent;
955        }
956        return y;
957    }
958
959    private Node predecessor(Node x) {
960        if (x == nil)
961            return nil;
962        if (x.left != nil) {
963            Node y = x.left;
964            while (y.right != nil)
965                y = y.right;
966            return y;
967        }
968        Node y = x.parent;
969        while (y != nil && x == y.left) {
970            x = y;
971            y = y.left;
972        }
973        return y;
974    }

976    /**
977     * Introduction to Algorithms (CLR) Second Edition
978     * 
979     * <pre>
980     * LEFT-ROTATE(T, x)
981     * y = right[x]                         Set y.
982     * right[x] = left[y]                   Turn y's left subtree into x's right subtree.
983     * if left[y] != nil[T]
984     *    then p[left[y]] = x
985     * p[y] = p[x]                          Link x's parent to y.
986     * if p[x] = nil[T]
987     *    then root[T] = y
988     *    else if x = left[p[x]]
989     *            then left[p[x]] = y
990     *            else right[p[x]] = y
991     * left[y] = x                          Put x on y's left.
992     * p[x] = y
993     * </pre>
994     */
995    private void leftRotate(final Node x) {
996        if (x != nil) {
997            Node n = x.right;
998            x.right = n.left;
999            if (n.left != nil)
1000                n.left.parent = x;
1001            n.parent = x.parent;
1002            if (x.parent == nil)
1003                root = n;
1004            else if (x.parent.left == x)
1005                x.parent.left = n;
1006            else
1007                x.parent.right = n;
1008            n.left = x;
1009            x.parent = n;
1010        }
1011    }
1012
1013    private void rightRotate(final Node x) {
1014        if (x != nil) {
1015            Node n = x.left;
1016            x.left = n.right;
1017            if (n.right != nil)
1018                n.right.parent = x;
1019            n.parent = x.parent;
1020            if (x.parent == nil)
1021                root = n;
1022            else if (x.parent.right == x)
1023                x.parent.right = n;
1024            else
1025                x.parent.left = n;
1026            n.right = x;
1027            x.parent = n;
1028        }
1029    }

1031    /**
1032     * Introduction to Algorithms (CLR) Second Edition
1033     * 
1034     * <pre>
1035     * RB-INSERT-FIXUP(T, z)
1036     * while color[p[z]] = RED
1037     *    do if p[z] = left[p[p[z]]]
1038     *          then y = right[p[p[z]]]
1039     *               if color[y] = RED
1040     *                  then color[p[z]] = BLACK                    Case 1
1041     *                       color[y] = BLACK                       Case 1 
1042     *                       color[p[p[z]]] = RED                   Case 1
1043     *                       z = p[p[z]]                            Case 1
1044     *                  else if z = right[p[z]]
1045     *                          then z = p[z]                       Case 2
1046     *                               LEFT-ROTATE(T, z)              Case 2
1047     *                       color[p[z]] = BLACK                    Case 3
1048     *                       color[p[p[z]]] = RED                   Case 3
1049     *                       RIGHT-ROTATE(T, p[p[z]])               Case 3
1050     *          else (same as then clause
1051     *                        with right and left exchanged)
1052     * color[root[T]] = BLACK
1053     * </pre>
1054     */
1055    private void fixAfterInsertion(Node z) {
1056        z.color = RED;
1057        while (z.parent.color == RED) {
1058            if (z.parent == z.parent.parent.left) {
1059                Node y = z.parent.parent.right;
1060                if (y.color == RED) {
1061                    z.parent.color = BLACK;
1062                    y.color = BLACK;
1063                    z.parent.parent.color = RED;
1064                    z = z.parent.parent;
1065                } else {
1066                    if (z == z.parent.right) {
1067                        z = z.parent;
1068                        leftRotate(z);
1069                    }
1070                    z.parent.color = BLACK;
1071                    z.parent.parent.color = RED;
1072                    rightRotate(z.parent.parent);
1073                }
1074            } else {
1075                Node y = z.parent.parent.left;
1076                if (y.color == RED) {
1077                    z.parent.color = BLACK;
1078                    y.color = BLACK;
1079                    z.parent.parent.color = RED;
1080                    z = z.parent.parent;
1081                } else {
1082                    if (z == z.parent.left) {
1083                        z = z.parent;
1084                        rightRotate(z);
1085                    }
1086                    z.parent.color = BLACK;
1087                    z.parent.parent.color = RED;
1088                    leftRotate(z.parent.parent);
1089                }
1090            }
1091        }
1092        root.color = BLACK;
1093    }

1095    /**
1096     * Introduction to Algorithms (CLR) Second Edition
1097     * 
1098     * <pre>
1099     * RB-DELETE-FIXUP(T, x)
1100     * while x != root[T] and color[x] = BLACK
1101     *    do if x = left[p[x]]
1102     *          then w = right[p[x]]
1103     *               if color[w] = RED
1104     *                  then color[w] = BLACK                               Case 1
1105     *                       color[p[x]] = RED                              Case 1
1106     *                       LEFT-ROTATE(T, p[x])                           Case 1
1107     *                       w = right[p[x]]                                Case 1
1108     *               if color[left[w]] = BLACK and color[right[w]] = BLACK
1109     *                  then color[w] = RED                                 Case 2
1110     *                       x = p[x]                                       Case 2
1111     *                  else if color[right[w]] = BLACK
1112     *                          then color[left[w]] = BLACK                 Case 3
1113     *                               color[w] = RED                         Case 3
1114     *                               RIGHT-ROTATE(T,w)                      Case 3
1115     *                               w = right[p[x]]                        Case 3
1116     *                       color[w] = color[p[x]]                         Case 4
1117     *                       color[p[x]] = BLACK                            Case 4
1118     *                       color[right[w]] = BLACK                        Case 4
1119     *                       LEFT-ROTATE(T, p[x])                           Case 4
1120     *                       x = root[T]                                    Case 4
1121     *          else (same as then clause with right and left exchanged)
1122     * color[x] = BLACK
1123     * </pre>
1124     */
1125    private void fixAfterDeletion(Node x) {
1126        while (x != root && x.color == BLACK) {
1127            if (x == x.parent.left) {
1128                Node w = x.parent.right;
1129                if (w.color == RED) {
1130                    w.color = BLACK;
1131                    x.parent.color = RED;
1132                    leftRotate(x.parent);
1133                    w = x.parent.right;
1134                }
1135                if (w.left.color == BLACK && w.right.color == BLACK) {
1136                    w.color = RED;
1137                    x = x.parent;
1138                } else {
1139                    if (w.right.color == BLACK) {
1140                        w.left.color = BLACK;
1141                        w.color = RED;
1142                        rightRotate(w);
1143                        w = x.parent.right;
1144                    }
1145                    w.color = x.parent.color;
1146                    x.parent.color = BLACK;
1147                    w.right.color = BLACK;
1148                    leftRotate(x.parent);
1149                    x = root;
1150                }
1151            } else {
1152                Node w = x.parent.left;
1153                if (w.color == RED) {
1154                    w.color = BLACK;
1155                    x.parent.color = RED;
1156                    rightRotate(w.parent);
1157                    w = x.parent.left;
1158                }
1159                if (w.right.color == BLACK && w.left.color == BLACK) {
1160                    w.color = RED;
1161                    x = x.parent;
1162                } else {
1163                    if (w.left.color == BLACK) {
1164                        w.right.color = BLACK;
1165                        w.color = RED;
1166                        leftRotate(w);
1167                        w = x.parent.left;
1168                    }
1169                    w.color = x.parent.color;
1170                    w.parent.color = BLACK;
1171                    w.left.color = BLACK;
1172                    rightRotate(x.parent);
1173                    x = root;
1174                }
1175            }
1176        }
1177        x.color = BLACK;
1178    }
