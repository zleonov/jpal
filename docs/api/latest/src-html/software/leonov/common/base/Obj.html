046    /**
047     * Returns a hash code value for the specified object, or 0 if {@code obj} is {@code null}.
048     * 
049     * @deprecated Java 7+ users should switch to {@link java.util.Objects#hashCode(Object)} instead.
050     * 
051     * @param obj the specified object
052     * @return a hash code value for the specified object, or 0 if {@code obj} is {@code null}
053     */
054    public static int hashCode(final Object obj) {
055        return obj == null ? 0 : obj.hashCode();
056    }

058    /**
059     * Returns the result of calling {@code toString()} on the specified object or "null" if the object is {@code null}.
060     * 
061     * @deprecated Java 7+ users should switch to {@link java.util.Objects#toString(Object)} instead.
062     * 
063     * @param obj the specified object
064     * @return the result of calling {@code toString()} on the specified object or "null" if the object is {@code null}
065     */
066    public static String toString(final Object obj) {
067        return obj == null ? "null" : obj.toString();
068    }

075     * 
076     * @param obj          the specified object

078     * @return the result of calling {@code toString()} on the specified object or {@code defaultValue} if the object is

085    /**
086     * Returns an order independent hash code for the specified objects.
087     * <p>
088     * <b>Warning:</b> do not confuse {@code unorderedHashCode(elements)} with {@code elements.hashCode()} which will return
089     * the hash code value of the {@code elements} object itself.
090     * 
091     * @param elements the specified objects
092     * @return an order independent hash code for the specified objects
093     */
094    public static int unorderedHashCode(final Iterable<?> elements) {
095        checkNotNull(elements, "elements == null");
096        return unorderedHashCode(elements.iterator());
097    }
098
099    /**
100     * Returns an order independent hash code for the specified objects.
101     * <p>
102     * <b>Warning:</b> do not confuse {@code unorderedHashCode(elements)} with {@code elements.hashCode()} which will return
103     * the hash code value of the {@code elements} object itself.
104     * 
105     * @param elements an {@code Iterator} over the specified objects
106     * @return an order independent hash code for the specified objects
107     */
108    public static int unorderedHashCode(final Iterator<?> elements) {
109        checkNotNull(elements, "elements == null");
110        int hashCode = 0;
111        while (elements.hasNext())
112            hashCode += Objects.hashCode(elements.next());
113        return hashCode;
114    }
115
116    /**
117     * Returns an order independent hash code for the specified objects.
118     * 
119     * @param elements the specified objects
120     * @return an order independent hash code for the specified objects
121     */
122    public static int unorderedHashCode(final Object... elements) {
123        checkNotNull(elements, "elements == null");
124        int hashCode = 0;
125        for (final Object obj : elements)
126            hashCode += Objects.hashCode(obj);
127        return hashCode;
128    }

142    /**
143     * Returns the first non-{@code null} value from the specified values or {@code null} if all values are {@code null}.
144     * 
145     * @param <T>    the type of value
146     * @param first  the first value
147     * @param second the second value
148     * @param rest   additional values
149     * @return the first non-{@code null} value from the specified values or {@code null} if all values are {@code null}
150     */
151    @SafeVarargs
152    public static <T> T coalesce(final T first, final T second, final T... rest) {
153        // return Streams.concat(Stream.of(first, second), Stream.of(rest)).filter(Objects::nonNull).findFirst().orElse(null);
154
155        if (first != null)
156            return first;
157
158        if (second != null)
159            return second;
160
161        if (rest != null)
162            for (final T value : rest)
163                if (value != null)
164                    return value;
165
166        return null;
167    }
168
169    /**
170     * Returns an {@code Optional} containing the first non-{@code null} value from the specified values or an
171     * {@link Optional#empty() empty} {@code Optional} if all values are {@code null}.
172     * 
173     * @param <T>    the type of value
174     * @param first  the first value
175     * @param second the second value
176     * @param rest   additional values
177     * @return an {@code Optional} containing the first non-{@code null} value from the specified values or an
178     *         {@link Optional#empty() empty} {@code Optional} if all values are {@code null}
179     */
180    @SafeVarargs
181    public static <T> Optional<T> firstNonNull(final T first, final T second, final T... rest) {
182        // return Streams.concat(Stream.of(first, second), Stream.of(rest)).filter(Objects::nonNull).findFirst();
183
184        return Optional.ofNullable(coalesce(first, second, rest));
185    }

187    /**
188     * Writes the specified object to the given file.
189     * <p>
190     * See {@link ObjectOutputStream#writeObject(Object)} for more information.
191     * <p>
192     * Exceptions are thrown for classes that should not be serialized. All exceptions are fatal. Any underlying streams
193     * will be closed before returning to the caller.
194     * 
195     * @param object the specified object
196     * @param path   the given file
197     * @throws InvalidClassException    if something is wrong with a class used by serialization
198     * @throws NotSerializableException if any object to be serialized does not implement the {@link java.io.Serializable}
199     *                                  interface
200     * @throws IOException              if an I/O error occurs
201     * @return the specified file
202     */
203    public static Path writeObject(final Object object, final Path path) throws IOException {
204        checkNotNull(object, "object == null");
205        checkNotNull(path, "path == null");
206
207        try (final ObjectOutputStream out = new ObjectOutputStream(Fs.newBufferedOutputStream(path, false))) {
208            out.writeObject(object);
209            return path;
210        }
211    }
212
213    /**
214     * Returns an object read from the specified file.
215     * <p>
216     * See {@link ObjectInputStream#readObject()} for more information.
217     * <p>
218     * Exceptions are thrown for classes that should not be deserialized. All exceptions are fatal. Any underlying streams
219     * will be closed before returning to the caller.
220     * 
221     * @param <T>  the type of object
222     * @param path the specified file
223     * @param type the class type of the object
224     * @throws ClassNotFoundException   if the class of the serialized object cannot be found
225     * @throws InvalidClassException    if something is wrong with a class used by serialization
226     * @throws IllegalArgumentException if the object is not assignable to the type {@code T}
227     * @throws IOException              if an I/O error occurs
228     * @return the object read from the specified file
229     * 
230     */
231    public static <T> T readObject(final Path path, final Class<? extends T> type) throws IOException, ClassNotFoundException {
232        checkNotNull(path, "path == null");
233        checkNotNull(type, "type == null");
234
235        try (final ObjectInputStream in = new ObjectInputStream(Fs.newBufferedInputStream(path))) {
236            final Object o = in.readObject();
237            checkArgument(type.isAssignableFrom(o.getClass()), "%s cannot be cast to %s", o.getClass(), type);
238            return type.cast(o);
239        }
240    }
