016package software.leonov.common.util;
017
018import static com.google.common.base.Preconditions.checkNotNull;
019
020import java.sql.Timestamp;
021import java.text.DateFormat;
022import java.text.SimpleDateFormat;
023import java.util.Date;
024
025/**
026 * Static utility methods for working with dates.
027 * 
028 * @deprecated Java 8+ users should switch to the new
029 *             <a target="_blank" href="https://docs.oracle.com/javase/tutorial/datetime/index.html">Date and Time
030 *             API</a>.
031 * 
032 * @author Zhenya Leonov
033 */
034public final class Dates {
035
036    private Dates() {
037    }
038
039    final private static DateFormat DATE_FORMAT = DateFormat.getDateTimeInstance(DateFormat.DEFAULT, DateFormat.DEFAULT);
040
041    /**
042     * Returns the current {@code Timestamp} formatted according to the specified pattern using {@link SimpleDateFormat}.
043     * <p>
044     * Exampe: {@code getFormattedTimestamp("YYYY-MM-DD")}
045     * 
046     * @param pattern the pattern to use when formatting the {@code Timestamp}
047     * @return the current {@code Timestamp} as a string
048     */
049    public static String getFormattedTimestamp(final String pattern) {
050        checkNotNull(pattern, "pattern == null");
051        return new SimpleDateFormat(pattern).format(new Date());
052    }
053
054    /**
055     * Returns a date/time formatter which uses the {@link DateFormat#DEFAULT DEFAULT} date and time formatting styles in
056     * the default locale.
057     * 
058     * @return a date/time formatter which uses the {@link DateFormat#DEFAULT DEFAULT} date and time formatting styles in
059     *         the default locale
060     */
061    public static DateFormat getDefaultDateTimeInstance() {
062        return DATE_FORMAT;
063    }
064
065    /**
066     * Converts a {@code java.util.Date} object to {@code java.sql.Timestamp}.
067     * 
068     * @param date the date to convert
069     * @return return a {@code java.sql.Timestamp} object converted from the given {@code java.util.Date} object
070     */
071    public static Timestamp getTimestamp(final Date date) {
072        checkNotNull(date, "date == null");
073        return new Timestamp(date.getTime());
074    }
075
076}
