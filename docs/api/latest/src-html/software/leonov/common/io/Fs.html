968    private static VisitResult walkFileTree(final File start, int depth, final int maxDepth, final FileWalker<?> walker) throws IOException {
969
970        File[] files = null;
971
972        try {
973            if (start.isDirectory() && (files = start.listFiles()) == null)
974                throw new IOException("cannot access " + start.getPath());
975            if (depth >= maxDepth || start.isFile())
976                return walker.visitFile(start);
977        } catch (final SecurityException ex) {
978            if (depth == 0)
979                throw ex;
980            return CONTINUE;
981        } catch (final IOException ex) {
982            return walker.visitFileFailed(start, ex);
983        }
984
985        VisitResult result = walker.preVisitDirectory(start);
986
987        if (result != CONTINUE)
988            return result;
989        for (final File file : files) {
990            if (!file.exists())
991                return walker.visitFileFailed(file, new FileNotFoundException(file.getPath()));
992            result = walkFileTree(file, depth + 1, maxDepth, walker);
993            if (result == SKIP_SIBLINGS)
994                break;
995            if (result == TERMINATE)
996                return TERMINATE;
997        }
998        return walker.postVisitDirectory(start);
999    }
