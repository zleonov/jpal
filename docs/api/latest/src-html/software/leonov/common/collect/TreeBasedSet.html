041/**
042 * A {@code Set} implementation based on a modified
043 * <a target="_blank" href="http://en.wikipedia.org/wiki/Red-black_tree">Red-Black Tree</a>. Elements are sorted from
044 * <i>least</i> to <i>greatest</i> according to their <i>natural ordering</i>, or by an explicit {@link Comparator}
045 * provided at creation. Attempting to insert {@code null} elements will succeed if the {@code Comparator} supports
046 * {@code null} values. Inserting non-comparable elements will result in a {@code ClassCastException}.
047 * <p>
048 * The iterators obtained from the {@link #iterator()} method are <i>fail-fast</i>. Attempts to modify the elements in
049 * this set at any time after an iterator is created, in any way except through the iterator's own remove method, will
050 * result in a {@code ConcurrentModificationException}.
051 * <p>
052 * This set is not <i>thread-safe</i>. If multiple threads modify this set concurrently it must be synchronized
053 * externally.
054 * <p>
055 * This implementation uses a comparator ({@link Ordering#natural() whether or not one is explicitly provided}) to
056 * perform all element comparisons. Two elements which are deemed equal by the comparator's {@code compare(E, E)} method
057 * are, from the standpoint of this set, equal.
058 * <p>
059 * The underlying Red-Black Tree provides the following worst case running time (where <i>n</i> is the size of this set
060 * and <i>m</i> is the size of the specified collection which is iterable in linear time):
061 * 
062 * <pre>
063 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
064 *   <tr>
065 *     <th style="text-align:center;">Method</th>
066 *     <th style="text-align:center;">Running Time</th>
067 *   </tr>
068 *   <tr>
069 *     <td>{@link #addAll(Collection) addAll(Collection)}<br>{@link #containsAll(Collection) containsAll(Collection)}<br>{@link #retainAll(Collection) retainAll(Collection)}<br>{@link #removeAll(Collection) removeAll(Collection)}</td>
070 *     <td style="text-align:center;"><i>O(m log n)</i></td>
071 *   </tr>
072 *   <tr>
073 *     <td>{@link #add(Object) add(E)}<br>{@link #contains(Object)}<br>{@link #remove(Object)}</td>
074 *     <td style="text-align:center;"><i>O(log n)</i></td>
075 *   </tr>
076 *   <tr>
077 *     <td>{@link #clear() clear()}<br>{@link #isEmpty() isEmpty()}<br>{@link #size()}<br>{@link Iterator#remove()}</td>
078 *     <td style="text-align:center;"><i>O(1)</i></td>
079 *   </tr>
080 * </table>
081 * </pre>
082 * 
083 * @deprecated <b>This class should only be used for testing. It provides no practical benefit when compared to
084 *             {@link java.util.TreeSet java.util.TreeSet} and lacks navigation methods defined in the
085 *             {@link NavigableSet} interface.</b>
086 * 
087 * @author Zhenya Leonov
088 * @param <E> the type of elements maintained by this set
089 */
090final public class TreeBasedSet<E> extends AbstractSet<E> implements SortedCollection<E>, Cloneable, Serializable {

124        return new TreeBasedSet<E>(comparator);

127    /**
128     * Creates a new {@code TreeBasedSet} containing the specified initial elements. If {@code elements} is an instance of
129     * {@link SortedSet}, {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or {@code SortedCollection}, this set will be
130     * ordered according to the same ordering. Otherwise, this set will be ordered according to the <i>natural ordering</i>
131     * of its elements.
132     * 
133     * @param elements the initial elements to be placed into the set
134     * @return a new {@code TreeBasedSet} containing the specified initial elements
135     * @throws ClassCastException   if any of the initial elements cannot be compared to one another according to this set's
136     *                              ordering
137     * @throws NullPointerException if any of the initial elements or {@code elements} itself is {@code null}
138     */
139    @SuppressWarnings({ "unchecked" })
140    public static <E> TreeBasedSet<E> create(final Iterable<? extends E> elements) {
141        checkNotNull(elements, "elements == null");
142
143        Comparator<? super E> comparator = null;
144        if (elements instanceof SortedSet<?>)
145            comparator = ((SortedSet<? super E>) elements).comparator();
146        else if (elements instanceof PriorityQueue<?>)
147            comparator = ((PriorityQueue<? super E>) elements).comparator();
148        else if (elements instanceof SortedCollection<?>)
149            comparator = ((SortedCollection<? super E>) elements).comparator();
150        else if (elements instanceof MinMaxPriorityQueue<?>)
151            comparator = ((MinMaxPriorityQueue<? super E>) elements).comparator();
152
153        if (comparator == null)
154            comparator = (Comparator<? super E>) Ordering.natural();
155
156        final TreeBasedSet<E> set = TreeBasedSet.create(comparator);
157        Iterables.addAll(set, elements);
158        return set;
159    }

172    /**

175    @Override

177//        checkNotNull(e, "e == null");
178        return insert(new Node(e));

181    @SuppressWarnings("unchecked")
182    @Override
183    public boolean contains(Object o) {
184        return search((E) o) != null;
185    }

187    @Override
188    public Iterator<E> iterator() {
189        return new Iterator<E>() {
190            private Node next = min;
191            private Node last = nil;
192            private int expectedModCount = modCount;
193
194            @Override
195            public boolean hasNext() {
196                return next != nil;
197            }
198
199            @Override
200            public void remove() {
201                checkForConcurrentModification();
202                checkState(last != nil);
203                if (last.left != nil && last.right != nil)
204                    next = last;
205                delete(last);
206                expectedModCount = modCount;
207                last = nil;
208            }
209
210            @Override
211            public E next() {
212                checkForConcurrentModification();
213                if (next == nil)
214                    throw new NoSuchElementException();
215                last = next;
216                next = successor(next);
217                return last.element;
218            }
219
220            private void checkForConcurrentModification() {
221                if (modCount != expectedModCount)
222                    throw new ConcurrentModificationException();
223            }
224        };
225    }
226
227    @SuppressWarnings("unchecked")
228    @Override
229    public boolean remove(Object o) {
230//        checkNotNull(o, "o == null");
231        final Node node = search((E) o);
232        if (node == null)
233            return false;
234        delete(node);
235        return true;
236    }

239    public int size() {
240        return size;

243    @Override
244    public void clear() {
245        modCount++;
246        root = min = nil;
247        size = 0;
248    }

250    /**
251     * Returns a shallow copy of this {@code TreeBasedSet}. The elements themselves are not cloned.
252     * 
253     * @return a shallow copy of this {@code TreeBasedSet}
254     */
255    @SuppressWarnings("unchecked")
256    @Override
257    public TreeBasedSet<E> clone() {
258        TreeBasedSet<E> clone;
259        try {
260            clone = (TreeBasedSet<E>) super.clone();
261        } catch (final CloneNotSupportedException e) {
262            throw new InternalError();
263        }
264        clone.modCount = clone.size = 0;
265        clone.root = clone.min = clone.min = clone.nil = new Node();
266        clone.addAll(this);
267        return clone;
268    }

270    private void writeObject(final ObjectOutputStream oos) throws IOException {
271        oos.defaultWriteObject();
272        oos.writeInt(size);
273        int actual = 0;
274
275        for (E e : this) {
276            oos.writeObject(e);
277            actual++;
278        }
279
280        if (actual != size)
281            System.out.println("size: " + size + " " + this);
282    }

284    @SuppressWarnings("unchecked")
285    private void readObject(final ObjectInputStream ois) throws IOException, ClassNotFoundException {
286        ois.defaultReadObject();
287        root = min = nil = new Node();
288        int size = ois.readInt();
289        for (int i = 0; i < size; i++)
290            add((E) ois.readObject());
291    }

297    static enum Color {
298        BLACK, RED;
299    }

301    private class Node {
302        private E element = null;
303        private Node parent, left, right;
304        private Color color = BLACK;
305
306        private Node() {
307            parent = this;
308            right = this;
309            left = this;
310        }
311
312        private Node(final E element) {
313            this.element = element;
314            parent = nil;
315            right = nil;
316            left = nil;
317        }
318    }

320    /**
321     * Introduction to Algorithms (CLR) Second Edition
322     * 
323     * <pre>
324     * RB-INSERT(T, z)
325     * y = nil[T]
326     * x = root[T]
327     * while x != nil[T]
328     *    do y = x
329     *       if key[z] < key[x]
330     *          then x = left[x]
331     *          else x = right[x]
332     * p[z] = y
333     * if y = nil[T]
334     *    then root[T] = z
335     *    else if key[z] < key[y]
336     *            then left[y] = z
337     *            else right[y] = z
338     * left[z] = nil[T]
339     * right[z] = nil[T]
340     * color[z] = RED
341     * RB-INSERT-FIXUP(T, z)
342     * </pre>
343     */
344    private boolean insert(final Node z) {
345        modCount++;
346        Node x = root;
347        Node y = nil;
348        while (x != nil) {
349            y = x;
350            int cmp = comparator.compare(z.element, x.element);
351            if (cmp == 0)
352                return false;
353            else if (cmp < 0)
354                x = x.left;
355            else
356                x = x.right;
357        }
358        z.parent = y;
359        if (y == nil)
360            root = z;
361        else if (comparator.compare(z.element, y.element) < 0)
362            y.left = z;
363        else
364            y.right = z;
365        fixAfterInsertion(z);
366        if (min == nil || comparator.compare(z.element, min.element) < 0)
367            min = z;
368        size++;
369        return true;
370    }

372    /**
373     * Introduction to Algorithms (CLR) Second Edition
374     * 
375     * <pre>
376     * RB-DELETE-FIXUP(T, z)
377     * if left[z] = nil[T] or right[z] = nil[T]
378     *    then y = z
379     *    else y = TREE-SUCCESSOR(z)
380     * if left[y] != nil[T]
381     *    then x = left[y]
382     *    else x = right[y]
383     * p[x] = p[y]
384     * if p[y] = nil[T]
385     *    then root[T] = x
386     *    else if y = left[p[y]]
387     *            then left[p[y]] = x
388     *            else right[p[y]] = x
389     * if y != z
390     *    then key[z] = key[y]
391     *         copy y's satellite data into z
392     * if color[y] = BLACK
393     *    then RB-DELETE-FIXUP(T, x)
394     * return y
395     * </pre>
396     */
397    private void delete(Node z) {
398        size--;
399        modCount++;
400        Node x, y;
401        if (min == z)
402            min = successor(z);
403        if (z.left == nil || z.right == nil)
404            y = z;
405        else
406            y = successor(z);
407        if (y.left != nil)
408            x = y.left;
409        else
410            x = y.right;
411        x.parent = y.parent;
412        if (y.parent == nil)
413            root = x;
414        else if (y == y.parent.left)
415            y.parent.left = x;
416        else
417            y.parent.right = x;
418        if (y != z)
419            z.element = y.element;
420        if (y.color == Color.BLACK)
421            fixAfterDeletion(x);
422    }
423
424    private Node search(final E e) {
425        Node n = root;
426        while (n != nil) {
427            int cmp = comparator.compare(e, n.element);
428            if (cmp == 0)
429                return n;
430            if (cmp < 0)
431                n = n.left;
432            else
433                n = n.right;
434        }
435        return null;
436    }

438    /**
439     * Introduction to Algorithms (CLR) Second Edition
440     * 
441     * <pre>
442     * TREE-SUCCESSOR(x)
443     * if right[x] != NIL
444     *    then return TREE-MINIMUM(right[x])
445     * y = p[x]
446     * while y != NIL and x = right[y]
447     *    do x = y
448     *       y = p[y]
449     * return y
450     * </pre>
451     */
452    private Node successor(Node x) {
453        if (x == nil)
454            return nil;
455        if (x.right != nil) {
456            Node y = x.right;
457            while (y.left != nil)
458                y = y.left;
459            return y;
460        }
461        Node y = x.parent;
462        while (y != nil && x == y.right) {
463            x = y;
464            y = y.parent;
465        }
466        return y;
467    }

469    /**
470     * Introduction to Algorithms (CLR) Second Edition
471     * 
472     * <pre>
473     * LEFT-ROTATE(T, x)
474     * y = right[x]                         Set y.
475     * right[x] = left[y]                   Turn y's left subtree into x's right subtree.
476     * if left[y] != nil[T]
477     *    then p[left[y]] = x
478     * p[y] = p[x]                          Link x's parent to y.
479     * if p[x] = nil[T]
480     *    then root[T] = y
481     *    else if x = left[p[x]]
482     *            then left[p[x]] = y
483     *            else right[p[x]] = y
484     * left[y] = x                          Put x on y's left.
485     * p[x] = y
486     * </pre>
487     */
488    private void leftRotate(final Node x) {
489        if (x != nil) {
490            Node n = x.right;
491            x.right = n.left;
492            if (n.left != nil)
493                n.left.parent = x;
494            n.parent = x.parent;
495            if (x.parent == nil)
496                root = n;
497            else if (x.parent.left == x)
498                x.parent.left = n;
499            else
500                x.parent.right = n;
501            n.left = x;
502            x.parent = n;
503        }
504    }
505
506    private void rightRotate(final Node x) {
507        if (x != nil) {
508            Node n = x.left;
509            x.left = n.right;
510            if (n.right != nil)
511                n.right.parent = x;
512            n.parent = x.parent;
513            if (x.parent == nil)
514                root = n;
515            else if (x.parent.right == x)
516                x.parent.right = n;
517            else
518                x.parent.left = n;
519            n.right = x;
520            x.parent = n;
521        }
522    }

524    /**
525     * Introduction to Algorithms (CLR) Second Edition
526     * 
527     * <pre>
528     * RB-INSERT-FIXUP(T, z)
529     * while color[p[z]] = RED
530     *    do if p[z] = left[p[p[z]]]
531     *          then y = right[p[p[z]]]
532     *               if color[y] = RED
533     *                  then color[p[z]] = BLACK                    Case 1
534     *                       color[y] = BLACK                       Case 1 
535     *                       color[p[p[z]]] = RED                   Case 1
536     *                       z = p[p[z]]                            Case 1
537     *                  else if z = right[p[z]]
538     *                          then z = p[z]                       Case 2
539     *                               LEFT-ROTATE(T, z)              Case 2
540     *                       color[p[z]] = BLACK                    Case 3
541     *                       color[p[p[z]]] = RED                   Case 3
542     *                       RIGHT-ROTATE(T, p[p[z]])               Case 3
543     *          else (same as then clause
544     *                        with right and left exchanged)
545     * color[root[T]] = BLACK
546     * </pre>
547     */
548    private void fixAfterInsertion(Node z) {
549        z.color = RED;
550        while (z.parent.color == RED) {
551            if (z.parent == z.parent.parent.left) {
552                Node y = z.parent.parent.right;
553                if (y.color == RED) {
554                    z.parent.color = BLACK;
555                    y.color = BLACK;
556                    z.parent.parent.color = RED;
557                    z = z.parent.parent;
558                } else {
559                    if (z == z.parent.right) {
560                        z = z.parent;
561                        leftRotate(z);
562                    }
563                    z.parent.color = BLACK;
564                    z.parent.parent.color = RED;
565                    rightRotate(z.parent.parent);
566                }
567            } else {
568                Node y = z.parent.parent.left;
569                if (y.color == RED) {
570                    z.parent.color = BLACK;
571                    y.color = BLACK;
572                    z.parent.parent.color = RED;
573                    z = z.parent.parent;
574                } else {
575                    if (z == z.parent.left) {
576                        z = z.parent;
577                        rightRotate(z);
578                    }
579                    z.parent.color = BLACK;
580                    z.parent.parent.color = RED;
581                    leftRotate(z.parent.parent);
582                }
583            }
584        }
585        root.color = BLACK;
586    }

588    /**
589     * Introduction to Algorithms (CLR) Second Edition
590     * 
591     * <pre>
592     * RB-DELETE-FIXUP(T, x)
593     * while x != root[T] and color[x] = BLACK
594     *    do if x = left[p[x]]
595     *          then w = right[p[x]]
596     *               if color[w] = RED
597     *                  then color[w] = BLACK                               Case 1
598     *                       color[p[x]] = RED                              Case 1
599     *                       LEFT-ROTATE(T, p[x])                           Case 1
600     *                       w = right[p[x]]                                Case 1
601     *               if color[left[w]] = BLACK and color[right[w]] = BLACK
602     *                  then color[w] = RED                                 Case 2
603     *                       x = p[x]                                       Case 2
604     *                  else if color[right[w]] = BLACK
605     *                          then color[left[w]] = BLACK                 Case 3
606     *                               color[w] = RED                         Case 3
607     *                               RIGHT-ROTATE(T,w)                      Case 3
608     *                               w = right[p[x]]                        Case 3
609     *                       color[w] = color[p[x]]                         Case 4
610     *                       color[p[x]] = BLACK                            Case 4
611     *                       color[right[w]] = BLACK                        Case 4
612     *                       LEFT-ROTATE(T, p[x])                           Case 4
613     *                       x = root[T]                                    Case 4
614     *          else (same as then clause with right and left exchanged)
615     * color[x] = BLACK
616     * </pre>
617     */
618    private void fixAfterDeletion(Node x) {
619        while (x != root && x.color == BLACK) {
620            if (x == x.parent.left) {
621                Node w = x.parent.right;
622                if (w.color == RED) {
623                    w.color = BLACK;
624                    x.parent.color = RED;
625                    leftRotate(x.parent);
626                    w = x.parent.right;
627                }
628                if (w.left.color == BLACK && w.right.color == BLACK) {
629                    w.color = RED;
630                    x = x.parent;
631                } else {
632                    if (w.right.color == BLACK) {
633                        w.left.color = BLACK;
634                        w.color = RED;
635                        rightRotate(w);
636                        w = x.parent.right;
637                    }
638                    w.color = x.parent.color;
639                    x.parent.color = BLACK;
640                    w.right.color = BLACK;
641                    leftRotate(x.parent);
642                    x = root;
643                }
644            } else {
645                Node w = x.parent.left;
646                if (w.color == RED) {
647                    w.color = BLACK;
648                    x.parent.color = RED;
649                    rightRotate(w.parent);
650                    w = x.parent.left;
651                }
652                if (w.right.color == BLACK && w.left.color == BLACK) {
653                    w.color = RED;
654                    x = x.parent;
655                } else {
656                    if (w.left.color == BLACK) {
657                        w.right.color = BLACK;
658                        w.color = RED;
659                        leftRotate(w);
660                        w = x.parent.left;
661                    }
662                    w.color = x.parent.color;
663                    w.parent.color = BLACK;
664                    w.left.color = BLACK;
665                    rightRotate(x.parent);
666                    x = root;
667                }
668            }
669        }
670        x.color = BLACK;
671    }
