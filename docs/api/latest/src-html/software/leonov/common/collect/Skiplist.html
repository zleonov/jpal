050/**
051 * A {@link Sortedlist} implementation based on a modified
052 * <a target="_blank" href="http://en.wikipedia.org/wiki/Skip_list">skip list</a>. Elements are sorted from <i>least</i>
053 * to <i>greatest</i> according to their <i>natural ordering</i>, or by an explicit {@link Comparator} provided at
054 * creation. Attempting to insert {@code null} elements will succeed if the {@code Comparator} supports {@code null}
055 * values. Inserting non-comparable elements will result in a {@code ClassCastException}.
056 * <p>
057 * The underlying array-based <a target="_blank" href="http://en.wikipedia.org/wiki/Skip_list">skip list</a> is modified
058 * to provide logarithmic running time for insertion, removal, and
059 * <a target="_blank" href="http://en.wikipedia.org/wiki/Random_access">random access</a> lookup operations (e.g. get
060 * the element at the i<i>th</i> index).
061 * <p>
062 * The iterators obtained from the {@link #iterator()} and {@link #listIterator()} methods are <i>fail-fast</i>.
063 * Attempts to modify the elements in this sorted-list at any time after an iterator is created, in any way except
064 * through the iterator's own remove method, will result in a {@code ConcurrentModificationException}. Further, the list
065 * iterator does not support the {@code add(E)} and {@code set(E)} operations.
066 * <p>
067 * This sorted-list is not <i>thread-safe</i>. If multiple threads modify this sorted-list concurrently it must be
068 * synchronized externally.
069 * <p>
070 * This implementation uses a comparator ({@link Ordering#natural() whether or not one is explicitly provided}) to
071 * perform all element comparisons. Two elements which are deemed equal by the comparator's {@code compare(E, E)} method
072 * are, from the standpoint of this sorted-list, equal. Further, no guarantee is made as to the final order of
073 * <i>equal</i> elements. Ties may be broken arbitrarily.
074 * <p>
075 * Invented by <a target="_blank" href="http://www.cs.umd.edu/~pugh/">Bill Pugh</a> in 1990, A skip list is a
076 * probabilistic data structure for maintaining items in sorted order. Strictly speaking it is impossible to make any
077 * hard guarantees regarding the worst-case performance of this class. Practical performance is <i>expected</i> to be
078 * logarithmic with an extremely high degree of probability as the skip list grows.
079 * <p>
080 * The following table summarizes the performance of this class compared to a {@link Treelist} (where n is the size of
081 * this sorted-list and <i>m</i> is the size of the specified collection which is iterable in linear time):
082 * 
083 * <pre>
084 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
085 *   <tr>
086 *     <th style="text-align:center;" rowspan="2">Method</th>
087 *     <th style="text-align:center;" colspan="2">Running Time</th>
088 *   </tr>
089 *   <tr>
090 *     <td style="text-align:center;"><b>Skiplist</b><br>(<i>expected</i>)</td>
091 *     <td style="text-align:center;"><b>Treelist</b><br>(<i>amortized</i>)</td>
092 *   </tr>
093 *   <tr>
094 *     <td>{@link #addAll(Collection) addAll(Collection)}<br>{@link #containsAll(Collection) containsAll(Collection)}<br>{@link #retainAll(Collection) retainAll(Collection)}<br>{@link #removeAll(Collection) removeAll(Collection)}</td>
095 *     <td style="text-align:center;" colspan="2"><i>O(m log n)</i></td>
096 *   </tr>
097 *   <tr>
098 *     <td>{@link #indexOf(Object)}<br>{@link #lastIndexOf(Object)}<br>{@link #get(int)}<br>{@link #remove(int)}<br>{@link #listIterator(int)}</td>
099 *     <td style="text-align:center;" bgcolor="FFCC99"><i>O(log n)</i></td>
100 *     <td style="text-align:center;" bgcolor="FFCCCC"><i>O(n)</i></td>
101 *   </tr>
102 *   <tr>
103 *     <td>{@link #add(Object) add(E)}<br>{@link #contains(Object)}<br>{@link #remove(Object)}</td>
104 *     <td style="text-align:center;" colspan="2"><i>O(log n)</i></td>
105 *   </tr>
106 *   <tr>
107 *     <td>{@link #clear() clear()}<br>{@link #isEmpty() isEmpty()}<br>{@link #size()}<br></td>
108 *     <td style="text-align:center;" colspan="2"><i>O(1)</i></td>
109 *   </tr>
110 *   <tr>
111 *     <td>{@link Iterator#remove()}<br>{@link ListIterator#remove()}</td>
112 *     <td style="text-align:center;" bgcolor="FFCC99"><i>O(log n)</i></td>
113 *     <td style="text-align:center;" bgcolor="FFCCCC"><i>O(1)</i></td>
114 *   </tr>
115 * </table>
116 * </pre>
117 * 
118 * The {@code subList} views exhibit identical time complexity, with the exception of the {@code clear()} operation
119 * which runs in linear time proportional to the size of the view.
120 * 
121 * @author Zhenya Leonov
122 * @param <E> the type of elements maintained by this sorted-list
123 * @see Treelist
124 */
125public class Skiplist<E> extends AbstractCollection<E> implements Sortedlist<E>, Serializable, Cloneable {

139    private Skiplist(final Comparator<? super E> comparator) {
140        this.comparator = comparator;
141        for (int i = 0; i < MAX_LEVEL; i++) {
142            head.next[i] = head;
143            head.dist[i] = 1;
144        }
145        head.prev = head;
146    }

160    /**
161     * Creates a new {@code Skiplist} containing the specified initial elements. If {@code elements} is an instance of
162     * {@link SortedSet}, {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or {@code SortedCollection}, this sorted-list
163     * will be ordered according to the same ordering. Otherwise, this sorted-list will be ordered according to the
164     * <i>natural ordering</i> of its elements.
165     * 
166     * @param <E>      the element type
167     * @param elements the collection whose elements are to be placed into the sorted-list
168     * @return a new {@code Skiplist} containing the elements of the specified collection
169     * @throws ClassCastException   if elements of the specified collection cannot be compared to one another according to
170     *                              this sorted-list's ordering
171     * @throws NullPointerException if any of the elements of the specified collection or the collection itself is
172     *                              {@code null}
173     */
174    @SuppressWarnings({ "unchecked" })
175    public static <E> Skiplist<E> create(final Iterable<? extends E> elements) {
176        checkNotNull(elements, "elements == null");
177
178        Comparator<? super E> comparator = null;
179        if (elements instanceof SortedSet<?>)
180            comparator = ((SortedSet<? super E>) elements).comparator();
181        else if (elements instanceof PriorityQueue<?>)
182            comparator = ((PriorityQueue<? super E>) elements).comparator();
183        else if (elements instanceof SortedCollection<?>)
184            comparator = ((SortedCollection<? super E>) elements).comparator();
185        else if (elements instanceof MinMaxPriorityQueue<?>)
186            comparator = ((MinMaxPriorityQueue<? super E>) elements).comparator();
187
188        if (comparator == null)
189            comparator = (Comparator<? super E>) Ordering.natural();
190
191        return orderedBy(comparator).create(elements);
192    }

272    @Override
273    public boolean add(E e) {
274//        checkNotNull(e, "e == null");
275        @SuppressWarnings("unchecked")
276        final Node<E>[] update   = new Node[MAX_LEVEL];
277        final int       newLevel = randomLevel();
278        Node<E>         x        = head;
279        Node<E>         y        = head;
280        int             i;
281        int             idx      = 0;
282        for (i = level - 1; i >= 0; i--) {
283            while (x.next[i] != y && comparator.compare(x.next[i].element, e) < 0) {
284                idx += x.dist[i];
285                x    = x.next[i];
286            }
287            y         = x.next[i];
288            update[i] = x;
289            index[i]  = idx;
290        }
291        if (newLevel > level) {
292            for (i = level; i < newLevel; i++) {
293                head.dist[i] = size + 1;
294                update[i]    = head;
295            }
296            level = newLevel;
297        }
298        x = new Node<E>(e, newLevel);
299        for (i = 0; i < level; i++) {
300            if (i > newLevel - 1)
301                update[i].dist[i]++;
302            else {
303                x.next[i]         = update[i].next[i];
304                update[i].next[i] = x;
305                x.dist[i]         = index[i] + update[i].dist[i] - idx;
306                update[i].dist[i] = idx + 1 - index[i];
307
308            }
309        }
310        x.prev        = update[0];
311        x.next().prev = x;
312        modCount++;
313        size++;
314        return true;
315    }

330    @SuppressWarnings("unchecked")
331    @Override
332    public int indexOf(Object o) {
333        if (o != null) {
334            Node<E> curr    = head;
335            int     idx     = 0;
336            final E element = (E) o;
337            for (int i = level - 1; i >= 0; i--)
338                while (curr.next[i] != head && comparator.compare(curr.next[i].element, element) < 0) {
339                    idx  += curr.dist[i];
340                    curr  = curr.next[i];
341                }
342            curr = curr.next();
343            if (curr != head && comparator.compare(curr.element, element) == 0)
344                return idx;
345        }
346        return -1;
347    }

351    public int lastIndexOf(Object o) {
352        if (o != null) {
353            Node<E> curr    = head;
354            int     idx     = -1;
355            final E element = (E) o;
356            for (int i = level - 1; i >= 0; i--)
357                while (curr.next[i] != head && comparator.compare(curr.next[i].element, element) <= 0) {
358                    idx  += curr.dist[i];
359                    curr  = curr.next[i];
360                }
361            if (curr != head && comparator.compare(curr.element, element) == 0)
362                return idx;
363        }
364        return -1;
365    }

377    @Override
378    public ListIterator<E> listIterator(final int index) {
379        checkPositionIndex(index, size);
380        return new ListIterator<E>() {
381            private Node<E> next             = (index == 0) ? head.next[0] : search(index);
382            private Node<E> last             = null;
383            private int     nextIndex        = index;
384            private int     expectedModCount = modCount;
385
386            @Override
387            public void add(E element) {
388                throw new UnsupportedOperationException();
389            }
390
391            @Override
392            public boolean hasNext() {
393                return nextIndex < size();
394            }
395
396            @Override
397            public boolean hasPrevious() {
398                return nextIndex > 0;
399            }
400
401            @Override
402            public E next() {
403                checkForConcurrentModification();
404                if (!hasNext())
405                    throw new NoSuchElementException();
406                nextIndex++;
407                last = next;
408                next = next.next[0];
409                return last.element;
410            }
411
412            @Override
413            public int nextIndex() {
414                return nextIndex;
415            }
416
417            @Override
418            public E previous() {
419                checkForConcurrentModification();
420                if (!hasPrevious())
421                    throw new NoSuchElementException();
422                last = next = next.prev;
423                nextIndex--;
424                return next.element;
425            }
426
427            @Override
428            public int previousIndex() {
429                return nextIndex - 1;
430            }
431
432            @Override
433            public void remove() {
434                checkForConcurrentModification();
435                checkState(last != null);
436                if (last == next.prev)
437                    Skiplist.this.remove(nextIndex-- - 1);
438                else {
439                    next = next.next[0];
440                    Skiplist.this.remove(nextIndex);
441                }
442                expectedModCount = modCount;
443                last             = null;
444            }
445
446            @Override
447            public void set(E element) {
448                throw new UnsupportedOperationException();
449            }
450
451            private void checkForConcurrentModification() {
452                if (expectedModCount != modCount)
453                    throw new ConcurrentModificationException();
454            }
455        };
456    }

458    @SuppressWarnings("unchecked")
459    @Override
460    public boolean remove(Object o) {
461//        if (o == null)
462//            return false;
463        final Node<E>[] update  = new Node[MAX_LEVEL];
464        final E         element = (E) o;
465        Node<E>         curr    = head;
466        for (int i = level - 1; i >= 0; i--) {
467            while (curr.next[i] != head && comparator.compare(curr.next[i].element, element) < 0)
468                curr = curr.next[i];
469            update[i] = curr;
470        }
471        curr = curr.next();
472        if (curr == head || comparator.compare(curr.element, element) != 0)
473            return false;
474        delete(curr, update);
475        return true;
476    }
477
478    @Override
479    public E remove(int index) {
480        checkElementIndex(index, size);
481        @SuppressWarnings("unchecked")
482        final Node<E>[] update = new Node[MAX_LEVEL];
483        Node<E>         curr   = head;
484        int             idx    = 0;
485        for (int i = level - 1; i >= 0; i--) {
486            while (idx + curr.dist[i] <= index) {
487                idx  += curr.dist[i];
488                curr  = curr.next[i];
489            }
490            update[i] = curr;
491        }
492        curr = curr.next();
493        delete(curr, update);
494        return curr.element;
495    }

521    @Override
522    public boolean equals(Object o) {
523        if (o == this)
524            return true;
525        if (!(o instanceof Sortedlist))
526            return false;
527        try {
528            @SuppressWarnings("unchecked")
529            final Iterator<E> i = ((Collection<E>) o).iterator();
530            for (E e : this)
531                if (comparator.compare(e, i.next()) != 0)
532                    return false;
533            return !i.hasNext();
534        } catch (final ClassCastException e) {
535            return false;
536        } catch (final NullPointerException e) {
537            return false;
538        }
539    }

547    /**
548     * Returns a shallow copy of this {@code Skiplist}. The elements themselves are not cloned.
549     * 
550     * @return a shallow copy of this {@code Skiplist}
551     */
552    @SuppressWarnings("unchecked")
553    @Override
554    public Skiplist<E> clone() throws CloneNotSupportedException {
555        Skiplist<E> clone;
556        try {
557            clone = (Skiplist<E>) super.clone();
558        } catch (final CloneNotSupportedException e) {
559            throw new InternalError();
560        }
561        for (int i = 0; i < MAX_LEVEL; i++) {
562            clone.head.next[i] = clone.head;
563            clone.head.dist[i] = 1;
564        }
565        clone.head.prev = clone.head;
566        clone.random    = ThreadLocalRandom.current();
567        clone.level     = 1;
568        clone.modCount  = 0;
569        clone.size      = 0;
570        clone.addAll(this);
571        return clone;
572    }

574    private void writeObject(final ObjectOutputStream oos) throws IOException {
575        oos.defaultWriteObject();
576        oos.writeInt(size);
577        for (E e : this)
578            oos.writeObject(e);
579    }

581    @SuppressWarnings("unchecked")
582    private void readObject(final ObjectInputStream ois) throws IOException, ClassNotFoundException {
583        ois.defaultReadObject();
584        head = new Node<E>(null, MAX_LEVEL);
585        for (int i = 0; i < MAX_LEVEL; i++) {
586            head.next[i] = head;
587            head.dist[i] = 1;
588        }
589        index     = new int[MAX_LEVEL];
590        head.prev = head;
591        random    = ThreadLocalRandom.current();
592        level     = 1;
593        int size = ois.readInt();
594        for (int i = 0; i < size; i++)
595            add((E) ois.readObject());
596    }

681    private static class Node<E> {
682        private E               element;
683        private Node<E>         prev;
684        private final Node<E>[] next;
685        private final int[]     dist;
686
687        @SuppressWarnings("unchecked")
688        private Node(final E element, final int size) {
689            this.element = element;
690            next         = new Node[size];
691            dist         = new int[size];
692        }
693
694        private Node<E> next() {
695            return next[0];
696        }
697    }

699    private int randomLevel() {
700        int randomLevel = 1;
701        while (randomLevel < MAX_LEVEL - 1 && random.nextDouble() < P)
702            randomLevel++;
703        return randomLevel;
704    }

706    private void delete(final Node<E> node, final Node<E>[] update) {
707        for (int i = 0; i < level; i++)
708            if (update[i].next[i] == node) {
709                update[i].next[i]  = node.next[i];
710                update[i].dist[i] += node.dist[i] - 1;
711            } else
712                update[i].dist[i]--;
713        node.next().prev = node.prev;
714        while (head.next[level - 1] == head && level > 1)
715            level--;
716        modCount++;
717        size--;
718    }
719
720    Node<E> search(final E element) {
721        Node<E> curr = head;
722        for (int i = level - 1; i >= 0; i--)
723            while (curr.next[i] != head && comparator.compare(curr.next[i].element, element) < 0)
724                curr = curr.next[i];
725        curr = curr.next();
726        if (curr != head && comparator.compare(curr.element, element) == 0)
727            return curr;
728        return null;
729    }
730
731    private Node<E> search(final int index) {
732        Node<E> curr = head;
733        int     idx  = -1;
734        for (int i = level - 1; i >= 0; i--)
735            while (idx + curr.dist[i] <= index) {
736                idx  += curr.dist[i];
737                curr  = curr.next[i];
738            }
739        return curr;
740    }

742    @SuppressWarnings("serial")
743    private final class Sublist extends Skiplist<E> {
744        private final Skiplist<E> list;
745        private int               offset;
746        private Node<E>           from;
747        private Node<E>           to;
748
749        public Sublist(final Skiplist<E> list, final int fromIndex, final int toIndex) {
750            super(list.comparator);
751            this.list     = list;
752            this.modCount = list.modCount;
753            offset        = fromIndex;
754            this.size     = toIndex - fromIndex;
755            from          = list.search(fromIndex);
756            to            = list.search(toIndex - 1);
757        }
758
759        @Override
760        public Skiplist<E> clone() throws CloneNotSupportedException {
761            throw new CloneNotSupportedException();
762        }
763
764        // do we need this?
765        private void writeObject(java.io.ObjectOutputStream out) throws IOException {
766            throw new NotSerializableException();
767        }
768
769        // do we need this?
770        private void readObject(final ObjectInputStream in) throws IOException, ClassNotFoundException {
771            throw new NotSerializableException();
772        }
773
774        private void checkForConcurrentModification() {
775            if (this.modCount != list.modCount)
776                throw new ConcurrentModificationException();
777        }
778
779        @Override
780        public int size() {
781            checkForConcurrentModification();
782            return this.size;
783        }
784
785        @Override
786        public boolean add(E e) {
787            checkForConcurrentModification();
788            checkNotNull(e, "e == null");
789            checkArgument(inRange(e, from, to));
790            list.add(e);
791            this.modCount = list.modCount;
792            this.size++;
793            return true;
794
795        }
796
797        @Override
798        public boolean remove(Object o) {
799            checkForConcurrentModification();
800            checkNotNull(o, "o == null");
801            @SuppressWarnings("unchecked")
802            E e = (E) o;
803            checkArgument(inRange(e, from, to));
804            if (comparator.compare(e, to.element) == 0) {
805                list.remove(size - 1 + offset);
806                to = to.prev;
807            } else
808                list.remove(e);
809            this.modCount = list.modCount;
810            this.size--;
811            return true;
812        }
813
814        @Override
815        public void clear() {
816            checkForConcurrentModification();
817            final Iterator<?> iterator = iterator();
818            while (iterator.hasNext()) {
819                iterator.next();
820                iterator.remove();
821            }
822        }
823
824        @Override
825        public E get(int index) {
826            checkForConcurrentModification();
827            checkElementIndex(index, this.size);
828            return list.get(index + offset);
829        }
830
831        @Override
832        public E remove(int index) {
833            checkForConcurrentModification();
834            checkElementIndex(index, this.size);
835            if (index == this.size - 1)
836                to = to.prev;
837            if (index == 0)
838                from = from.next();
839            final E e = list.remove(index + offset);
840            this.modCount = list.modCount;
841            this.size--;
842            return e;
843        }
844
845        @Override
846        public int indexOf(Object o) {
847            checkForConcurrentModification();
848            checkNotNull(o, "o == null");
849            @SuppressWarnings("unchecked")
850            E e = (E) o;
851            if (!inRange(e, from, to))
852                return -1;
853            if (comparator.compare(e, from.element) == 0)
854                return 0;
855            final int result = list.indexOf(e);
856            return result == -1 ? -1 : result - offset;
857        }
858
859        @Override
860        public int lastIndexOf(Object o) {
861            checkForConcurrentModification();
862            checkNotNull(o, "o == null");
863            @SuppressWarnings("unchecked")
864            E e = (E) o;
865            if (!inRange(e, from, to))
866                return -1;
867            if (comparator.compare(to.element, e) == 0)
868                return this.size - 1;
869            final int result = list.lastIndexOf(e);
870            return result == -1 ? -1 : result - offset;
871        }
872
873        @Override
874        public ListIterator<E> listIterator(final int index) {
875            checkForConcurrentModification();
876            checkPositionIndex(index, Sublist.this.size);
877            return new ListIterator<E>() {
878                final ListIterator<E> li = list.listIterator(index + offset);
879
880                @Override
881                public boolean hasNext() {
882                    return nextIndex() < Sublist.this.size;
883                }
884
885                @Override
886                public E next() {
887                    if (hasNext())
888                        return li.next();
889                    else
890                        throw new NoSuchElementException();
891                }
892
893                @Override
894                public boolean hasPrevious() {
895                    return previousIndex() >= 0;
896                }
897
898                public E previous() {
899                    if (hasPrevious())
900                        return li.previous();
901                    else
902                        throw new NoSuchElementException();
903                }
904
905                public int nextIndex() {
906                    return li.nextIndex() - offset;
907                }
908
909                public int previousIndex() {
910                    return li.previousIndex() - offset;
911                }
912
913                @Override
914                public void remove() {
915                    li.remove();
916                    Sublist.this.modCount = list.modCount;
917                    Sublist.this.size--;
918                }
919
920                @Override
921                public void set(E e) {
922                    throw new UnsupportedOperationException();
923                }
924
925                @Override
926                public void add(E e) {
927                    throw new UnsupportedOperationException();
928                }
929
930            };
931        }
932
933        @Override
934        Node<E> search(final E e) {
935            checkForConcurrentModification();
936            if (!inRange(e, from, to))
937                return null;
938            if (comparator.compare(e, from.element) == 0)
939                return from;
940            if (comparator.compare(e, to.element) == 0)
941                return to;
942            return list.search(e);
943        }
944
945        private boolean inRange(final E e, final Node<E> from, final Node<E> to) {
946            return (comparator.compare(from.element, e) < 1 && comparator.compare(e, to.element) < 1);
947        }
